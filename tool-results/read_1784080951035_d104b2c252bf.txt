    26→/* ---------------- Build Trace Recorder ---------------- */
    27→//
    28→// Records a per-task runtime trace that PROVES the ExecutionEngine actually
    29→// performs parallel scheduling, dependency resolution, and deterministic
    30→// completion ordering at runtime. Each task gets one TraceEntry with:
    31→//   - scheduledAt: ts when submit()/submitAll() registered the task
    32→//   - startedAt:   ts when the executor actually picked it up
    33→//   - completedAt: ts when complete() finished (success or failure)
    34→//   - parallelBatch: wave number — increments when the scheduler dispatches
    35→//     2+ tasks in the same trySchedule() tick OR after a >50ms gap (which
    36→//     indicates the previous wave's async work finished and a new wave is
    37→//     starting). Tasks in the same batch are the ones that ran in parallel.
    38→//
    39→// The trace is exposed via executionEngine.getTrace() and the
    40→// /api/build/trace HTTP endpoint for runtime inspection.
    41→
    42→export interface TraceEntry {
    43→  taskId: string;
    44→  taskTitle: string;
    45→  agent: string; // task.agent
    46→  stageId: string;
    47→  dependsOn: string[]; // task.dependsOn at scheduling time
    48→  scheduledAt: number; // ms timestamp when submit()/submitAll() was called
    49→  startedAt: number | null; // when executor actually picked it up
    50→  completedAt: number | null;
    51→  status: "pending" | "running" | "completed" | "failed";
    52→  parallelBatch: number; // 1, 2, 3... — wave number (see above)
    53→}
    54→
    55→export class BuildTrace {
    56→  private entries = new Map<string, TraceEntry>();
    57→  private currentBatch = 0;
    58→  private lastDispatchTs = 0;
    59→  // 50ms gap between scheduler dispatches is treated as a wave boundary —
    60→  // short enough that synchronous submitAll() bursts stay in one batch,
    61→  // long enough that real async tool completions (which take ≥1 tick) start
    62→  // a new batch when they unblock dependent tasks.
    63→  private static readonly WAVE_GAP_MS = 50;
    64→
    65→  /** Record that a task was scheduled (called from submit()). */
    66→  recordScheduled(task: Task): void {
    67→    if (this.entries.has(task.id)) return;
    68→    this.entries.set(task.id, {
    69→      taskId: task.id,
    70→      taskTitle: task.title,
    71→      agent: task.agent,
    72→      stageId: task.stageId,
    73→      dependsOn: [...task.dependsOn],
    74→      scheduledAt: Date.now(),
    75→      startedAt: null,
    76→      completedAt: null,
    77→      status: "pending",
    78→      parallelBatch: 0, // assigned when the task actually starts
    79→    });
    80→  }
    81→
    82→  /**
    83→   * Decide the parallelBatch number for a new scheduler wave.
    84→   * Called ONCE per trySchedule() call that actually dispatches ≥1 task.
    85→   * `tasks` = the tasks about to start in this tick (used to detect new
    86→   * waves triggered by synchronous in-memory completions, where the time
    87→   * gap is 0ms but a dep that just completed proves a wave boundary).
    88→   * Returns the batch number to assign to all of them.
    89→   */
    90→  nextBatch(tasks: Task[]): number {
    91→    const now = Date.now();
    92→    const gap = now - this.lastDispatchTs;
    93→    // New wave when:
    94→    //   - first ever dispatch (currentBatch === 0)
    95→    //   - 2+ tasks dispatched in the same scheduler tick (true parallel batch)
    96→    //   - significant time gap since the previous dispatch (>50ms) — previous
    97→    //     wave's async work finished and a new wave is starting
    98→    //   - any task being dispatched has a dependency that completed in the
    99→    //     CURRENT batch — proves a synchronous in-memory cascade where a
   100→    //     dep finished and its dependent is now starting (a new wave even
   101→    //     though Date.now() hasn't advanced)
   102→    const hasDepInCurrentBatch = tasks.some((t) =>
   103→      t.dependsOn.some((depId) => {
   104→        const dep = this.entries.get(depId);
   105→        return (
   106→          !!dep &&
   107→          dep.parallelBatch === this.currentBatch &&
   108→          dep.status === "completed"
   109→        );
   110→      })
   111→    );
   112→    const isNewWave =
   113→      this.currentBatch === 0 ||
   114→      tasks.length >= 2 ||
   115→      gap > BuildTrace.WAVE_GAP_MS ||
   116→      hasDepInCurrentBatch;
   117→    if (isNewWave) {
   118→      this.currentBatch++;
   119→    }
   120→    this.lastDispatchTs = now;
   121→    return this.currentBatch;
   122→  }
   123→
   124→  /** Record that a task transitioned to running. */
   125→  recordStarted(task: Task, batch: number): void {
   126→    const e = this.entries.get(task.id);
   127→    if (!e) return;
   128→    e.startedAt = Date.now();
   129→    e.status = "running";
   130→    e.parallelBatch = batch;
   131→  }
   132→
   133→  /** Record that a task finished (success or failure). */
   134→  recordCompleted(task: Task, status: "completed" | "failed"): void {
   135→    const e = this.entries.get(task.id);
   136→    if (!e) return;
   137→    e.completedAt = Date.now();
   138→    e.status = status;
   139→  }
   140→
   141→  /** Full trace sorted by scheduledAt then startedAt (stable, deterministic). */
   142→  getTrace(): TraceEntry[] {
   143→    return [...this.entries.values()].sort((a, b) => {
   144→      if (a.scheduledAt !== b.scheduledAt) return a.scheduledAt - b.scheduledAt;
   145→      const as = a.startedAt ?? 0;
   146→      const bs = b.startedAt ?? 0;
   147→      if (as !== bs) return as - bs;
   148→      return a.taskId.localeCompare(b.taskId);
   149→    });
   150→  }
   151→
   152→  /** Reset the trace (called from ExecutionEngine.reset()). */
   153→  clear(): void {
   154→    this.entries.clear();
   155→    this.currentBatch = 0;
   156→    this.lastDispatchTs = 0;
   157→  }
   158→}
   159→
   160→export class ExecutionEngine {
   161→  private tasks = new Map<string, Task>();
   162→  private queue: string[] = []; // task ids ready/queued
   163→  private running = new Set<string>();
   164→  private listeners = new Set<EventListener>();
   165→  private timers = new Map<string, ReturnType<typeof setTimeout>>();
   166→  private maxParallel: number;
   167→  private onTaskDone?: (task: Task) => void;
   168→  private cancelled = false;
   169→  private eventCounter = 0;
   170→  private trace = new BuildTrace();
   171→
   172→  constructor(opts: ExecutionEngineOptions = { maxParallel: 4 }) {
   173→    this.maxParallel = opts.maxParallel;
   174→    this.onTaskDone = opts.onTaskDone;
   175→  }
   176→
   177→  /* ---------------- Event Bus ---------------- */
   178→  subscribe(fn: EventListener): () => void {
   179→    this.listeners.add(fn);
   180→    return () => this.listeners.delete(fn);
   181→  }
   182→  private emit(partial: Omit<EngineEvent, "id" | "ts">): void {
   183→    const e: EngineEvent = { id: `ev-${++this.eventCounter}`, ts: Date.now(), ...partial };
   184→    this.listeners.forEach((fn) => fn(e));
   185→  }
   186→
   187→  /* ---------------- Task Queue + Dependency Scheduler ---------------- */
   188→  submit(task: Task): void {
   189→    this.tasks.set(task.id, task);
   190→    // Record the task in the build trace BEFORE any scheduling — this captures
   191→    // the scheduledAt timestamp at the moment the task entered the engine.
   192→    this.trace.recordScheduled(task);
   193→    task.status = task.dependsOn.length === 0 ? "ready" : "queued";
   194→    this.emit({ type: "task-queued", taskId: task.id, workflowId: task.workflowId, message: `Queued: ${task.title}`, level: "debug" });
   195→    this.trySchedule();
   196→  }
   197→
   198→  submitAll(tasks: Task[]): void {
   199→    for (const t of tasks) this.submit(t);
   200→  }
   201→
   202→  private trySchedule(): void {
   203→    if (this.cancelled) {
   204→      this.emit({ type: "task-failed", message: `trySchedule: cancelled=true, skipping`, level: "debug" });
   205→      return;
   206→    }
   207→    // Promote queued tasks whose dependencies are satisfied
   208→    let promoted = 0;
   209→    for (const t of this.tasks.values()) {
   210→      if (t.status === "queued" && this.depsSatisfied(t)) {
   211→        t.status = "ready";
   212→        promoted++;
   213→      }
   214→    }
   215→    // Collect ready tasks up to the parallelism limit. We snapshot the
   216→    // dispatch list BEFORE calling start() so that all tasks dispatched in
   217→    // this scheduler tick can be tagged with the same parallelBatch number.
   218→    const ready = [...this.tasks.values()].filter((t) => t.status === "ready");
   219→    const toDispatch: Task[] = [];
   220→    for (const t of ready) {
   221→      if (this.running.size + toDispatch.length >= this.maxParallel) break;
   222→      toDispatch.push(t);
   223→    }
   224→    if (toDispatch.length > 0) {
   225→      const batch = this.trace.nextBatch(toDispatch);
   226→      for (const t of toDispatch) {
   227→        this.start(t, batch);
   228→      }
   229→    }
   230→    if (promoted > 0 || ready.length > 0) {
   231→      this.emit({ type: "task-queued", message: `trySchedule: promoted ${promoted}, started ${toDispatch.length}, running=${this.running.size}`, level: "debug" });
   232→    }
   233→  }
   234→
   235→  private depsSatisfied(t: Task): boolean {
   236→    return t.dependsOn.every((depId) => {
   237→      const dep = this.tasks.get(depId);
   238→      return dep && (dep.status === "succeeded" || dep.status === "skipped");
   239→    });
   240→  }
   241→
   242→  /* ---------------- Parallel Execution Manager ---------------- */
   243→  // REAL execution: no setTimeout. Tasks either invoke a real tool (via the
   244→  // server-side ToolManager through the client bridge) or, for non-tool tasks,
   245→  // complete immediately with a real measured duration. Gate tasks run real
   246→  // tsc/eslint and self-heal via the LLM repair API on failure.
   247→  private start(task: Task, batch: number): void {
   248→    task.status = "running";
   249→    task.startedAt = Date.now();
   250→    this.running.add(task.id);
   251→    // Record the start (with the wave's parallelBatch number) AFTER the
   252→    // task is marked running but BEFORE the task-started event fires, so
   253→    // any subscriber that reads getTrace() sees the running state.
   254→    this.trace.recordStarted(task, batch);
   255→    this.emit({ type: "task-started", taskId: task.id, stageId: task.stageId, workflowId: task.workflowId, message: `Started: ${task.title}`, level: "info" });
   256→
   257→    // Fire-and-forget async completion. The duration is measured from real
   258→    // tool execution (or ~0ms for in-memory tasks), NOT a fake timeout.
   259→    void this.complete(task);
   260→  }
   261→
   262→  private async complete(task: Task): Promise<void> {
   263→    if (this.cancelled) return;
   264→    this.running.delete(task.id);
   265→    task.finishedAt = Date.now();
   266→    task.durationMs = task.finishedAt - (task.startedAt ?? task.finishedAt);
   267→
   268→    // If the task has a toolId, invoke the real tool now.
   269→    if (task.toolId && !task.gate) {
   270→      try {
   271→        const { invokeToolClient } = await import("./tool-client");
   272→        const cwd = (task as Task & { args?: { cwd?: string } }).args?.cwd;
   273→        const result = await invokeToolClient(task.toolId, cwd ? { cwd } : {});
   274→        task.result = result.success ? "ok" : `exit ${result.exitCode}`;
   275→        if (!result.success) {
   276→          task.status = "failed";
   277→          this.trace.recordCompleted(task, "failed");
   278→          this.emit({ type: "task-failed", taskId: task.id, stageId: task.stageId, message: `${task.title} failed (exit ${result.exitCode}): ${result.stderr.slice(0, 200)}`, level: "error" });
   279→          this.trySchedule();
   280→          return;
   281→        }
   282→        this.emit({ type: "task-succeeded", taskId: task.id, stageId: task.stageId, workflowId: task.workflowId, message: `${task.title} done in ${result.durationMs}ms (real tool run)`, level: "success" });
   283→      } catch (err) {
   284→        task.status = "failed";
   285→        this.trace.recordCompleted(task, "failed");
   286→        this.emit({ type: "task-failed", taskId: task.id, stageId: task.stageId, message: `${task.title} tool error: ${String(err)}`, level: "error" });
   287→        this.trySchedule();
   288→        return;
   289→      }
   290→    }
   291→
   292→    // Quality gate evaluation (if this task is a gate)
   293→    if (task.gate) {
   294→      const gateCtx = (task as Task & { gateContext?: import("./self-healing").GateEvaluationContext }).gateContext ?? {};
   295→      const result = await this.runGateWithHealing(task, gateCtx);
   296→      if (!result.passed) {
   297→        task.status = "failed";
   298→        this.trace.recordCompleted(task, "failed");
   299→        this.emit({ type: "task-failed", taskId: task.id, stageId: task.stageId, message: `Gate failed: ${task.gate} — ${result.detail}`, level: "warn" });
   300→        this.trySchedule();
   301→        return;
   302→      }
   303→      this.emit({ type: "gate-evaluated", taskId: task.id, stageId: task.stageId, message: `Gate passed: ${task.gate} (${result.metric})`, level: "success" });
   304→    }
   305→
   306→    task.status = "succeeded";
   307→    if (!task.result) task.result = "ok";
   308→    this.trace.recordCompleted(task, "completed");
   309→    this.emit({ type: "task-succeeded", taskId: task.id, stageId: task.stageId, workflowId: task.workflowId, message: `Done: ${task.title} (${task.durationMs}ms)`, level: "success" });
   310→    this.onTaskDone?.(task);
   311→    this.trySchedule();
   312→  }
   313→
   314→  /* ---------------- Retry Manager + Self-healing ---------------- */
   315→  // REAL self-healing: on gate failure, escalate through levels. For
   316→  // compilation/lint failures, call the LLM repair API (/api/repair) which
   317→  // returns a patched file, write it to the workspace via /api/workspace,
   318→  // then re-run the gate. No forced pass — if healing is exhausted, the
   319→  // gate genuinely fails.
   320→  private async runGateWithHealing(
   321→    task: Task,
   322→    ctx: import("./self-healing").GateEvaluationContext
   323→  ): Promise<GateResult> {
   324→    let result = await evaluateGate(task.gate!, ctx);
   325→    if (result.passed) return result;
   326→
   327→    let level: SelfHealLevel = "fastfix";
   328→    let attempts = 0;
   329→    const maxAttempts = 3;
   330→    while (attempts < maxAttempts) {
   331→      attempts++;
   332→      const next = selfHealController.nextLevel(task.id, level);
   333→      if (!next) break;
   334→      level = next;
   335→      this.emit({ type: "task-retried", taskId: task.id, message: `Self-heal @ ${SELF_HEAL_LEVELS.find((l) => l.id === level)?.label}`, level: "warn" });
   336→
   337→      // For compilation failures with a workspace, attempt an LLM repair.
   338→      if (task.gate === "compilation" && ctx.workspacePath && level !== "human-question") {
   339→        const repaired = await this.attemptLLMRepair(ctx.workspacePath);
   340→        if (repaired) {
   341→          // Diff already logged to Build Memory in attemptLLMRepair
   342→          void repaired.diff;
   343→        }
   344→      }
   345→
   346→      const retry = await evaluateGate(task.gate!, ctx);
   347→      if (retry.passed) {
   348→        return retry;
   349→      }
   350→      result = retry;
   351→    }
   352→    // Healing exhausted — genuinely fail (no force-pass).
   353→    return {
   354→      gate: task.gate!,
   355→      passed: false,
   356→      detail: `${task.gate} failed after ${attempts} self-heal attempts: ${result.detail}`,
   357→      metric: result.metric ?? "failed",
   358→    };
   359→  }
   360→
   361→  /** Attempt to repair compilation errors via the LLM repair API.
   362→   *  Reads the failing file, calls /api/repair, writes the patched content
   363→   *  back to the workspace, and logs the diff to Build Memory. */
   364→  private async attemptLLMRepair(workspacePath: string): Promise<{ file: string; diff: string } | null> {
   365→    try {
   366→      // Re-run tsc to get fresh errors
   367→      const { invokeToolClient } = await import("./tool-client");
   368→      const tscResult = await invokeToolClient("tsc-no-emit", { cwd: workspacePath });
   369→      if (tscResult.success || !tscResult.errors?.length) return null;
   370→
   371→      // Take the first error's file, read it, repair it, write it back.
   372→      const firstError = tscResult.errors[0];
   373→      const fileRes = await fetch(`/api/workspace?path=${encodeURIComponent(workspacePath)}&file=${encodeURIComponent(firstError.file)}`);
   374→      if (!fileRes.ok) return null;
   375→      const { content: originalContent } = await fileRes.json();
   376→
   377→      const repairRes = await fetch("/api/repair", {
   378→        method: "POST",
   379→        headers: { "Content-Type": "application/json" },
   380→        body: JSON.stringify({
   381→          filePath: firstError.file,
   382→          fileContent: originalContent,
   383→          errors: tscResult.errors,
   384→          language: "typescript",
   385→        }),
   386→      });
   387→      if (!repairRes.ok) return null;
   388→      const { patchedContent, tokensUsed } = await repairRes.json();
   389→
   390→      // Write the patched file back to the workspace
   391→      const writeRes = await fetch("/api/workspace", {
   392→        method: "PATCH",
   393→        headers: { "Content-Type": "application/json" },
   394→        body: JSON.stringify({ path: workspacePath, file: firstError.file, content: patchedContent }),
   395→      });
   396→      if (!writeRes.ok) return null;
   397→
   398→      // Compute a simple diff for observability + Build Memory
   399→      const origLines = originalContent.split("\n");
   400→      const newLines = patchedContent.split("\n");
   401→      const diffParts: string[] = [];
   402→      const maxLines = Math.max(origLines.length, newLines.length);
   403→      for (let i = 0; i < maxLines; i++) {
   404→        if (origLines[i] !== newLines[i]) {
   405→          if (origLines[i] !== undefined) diffParts.push(`- L${i + 1}: ${origLines[i]}`);
   406→          if (newLines[i] !== undefined) diffParts.push(`+ L${i + 1}: ${newLines[i]}`);
   407→        }
   408→      }
   409→      const diff = diffParts.join("\n").slice(0, 2000); // cap at 2KB
   410→
   411→      // Log the diff to Build Memory via projectMemory
   412→      const { projectMemory } = await import("./memories");
   413→      projectMemory.write(
   414→        "build",
   415→        `Repair diff: ${firstError.file}`,
   416→        JSON.stringify({
   417→          file: firstError.file,
   418→          errors: tscResult.errors.slice(0, 5).map((e) => ({ line: e.line, message: e.message })),
   419→          diff,
   420→          repairedAt: new Date().toISOString(),
   421→        }, null, 2),
   422→        "debugger"
   423→      );
   424→
   425→      // Emit an event for the UI/logs
   426→      this.emit({
   427→        type: "task-retried",
   428→        message: `Self-heal: patched ${firstError.file} (${diffParts.length} lines changed)`,
   429→        level: "info",
   430→      });
   431→
   432→      // Charge real tokens from the repair LLM call
   433→      const { observability } = await import("./observability");
   434→      observability.chargeTokens("debugger" as AgentRole, tokensUsed ?? 0, "new-project");
   435→
   436→      return { file: firstError.file, diff };
   437→    } catch {
   438→      return null;
   439→    }
   440→  }
   441→
   442→  /* ---------------- Cancellation Manager ---------------- */
   443→  cancelAll(): void {
   444→    this.cancelled = true;
   445→    for (const timer of this.timers.values()) clearTimeout(timer);
   446→    this.timers.clear();
   447→    for (const t of this.tasks.values()) {
   448→      if (t.status === "running" || t.status === "ready" || t.status === "queued") {
   449→        // Mark any in-flight task as failed in the trace so the runtime
   450→        // record reflects that it did not complete.
   451→        if (t.status === "running") this.trace.recordCompleted(t, "failed");
   452→        t.status = "cancelled";
   453→      }
   454→    }
   455→    this.running.clear();
   456→  }
   457→
   458→  reset(): void {
   459→    this.cancelAll();
   460→    this.cancelled = false;
   461→    this.tasks.clear();
   462→    this.queue = [];
   463→    this.running.clear();
   464→    this.trace.clear();
   465→  }
   466→
   467→  /* ---------------- Introspection ---------------- */
   468→  /**
   469→   * Runtime task-graph trace: one TraceEntry per task with scheduledAt,
   470→   * startedAt, completedAt, status, and parallelBatch (wave number). Used to
   471→   * PROVE parallel scheduling, dependency resolution, and completion ordering
   472→   * actually happen at runtime. Exposed via /api/build/trace.
   473→   */
   474→  getTrace(): TraceEntry[] {
   475→    return this.trace.getTrace();
   476→  }
   477→
   478→  allTasks(): Task[] {
   479→    return [...this.tasks.values()];
   480→  }
   481→  tasksForStage(stageId: string): Task[] {
   482→    return this.allTasks().filter((t) => t.stageId === stageId);
   483→  }
   484→  /**
   485→   * Coarse-grained stage status derived from the underlying tasks.
   486→   *
   487→   * Returns one of the UI-level `StageStatus` values (`"pending" |
   488→   * `"running" | "done" | "failed"`) — NOT a strict `TaskStatus`. The
   489→   * return type intentionally includes `"done"` and `"pending"`, which
   490→   * are stage-level concepts that don't exist on individual tasks.
   491→   */
   492→  stageStatus(stageId: string): TaskStatus | "pending" | "done" {
   493→    const tasks = this.tasksForStage(stageId);
   494→    if (tasks.length === 0) return "pending";
   495→    if (tasks.some((t) => t.status === "running")) return "running";
   496→    if (tasks.every((t) => t.status === "succeeded" || t.status === "skipped")) return "done";
   497→    if (tasks.some((t) => t.status === "failed" && t.retryLevel === undefined)) return "failed";
   498→    if (tasks.some((t) => t.status === "ready" || t.status === "queued")) return "running";
   499→    return "running";
   500→  }
   501→  isIdle(): boolean {
   502→    return this.running.size === 0 && ![...this.tasks.values()].some((t) => t.status === "ready" || t.status === "queued");
   503→  }
   504→
   505→  /**
   506→   * Compute progress as { completed, total, percent } from real task states.
   507→   * Used by the UI to drive the progress bar — no timers, no fake counts.
   508→   */
   509→  getProgress(): { completed: number; total: number; percent: number } {
   510→    const all = this.allTasks();
   511→    const total = all.length;
   512→    const completed = all.filter((t) => t.status === "succeeded" || t.status === "skipped").length;
   513→    const percent = total > 0 ? Math.round((completed / total) * 100) : 0;
   514→    return { completed, total, percent };
   515→  }
   516→
   517→  /**
   518→   * Derive the UI stage status for all 8 pipeline stages from real task states.
   519→   * Returns a map stageId → "pending" | "running" | "done" | "failed".
   520→   */
   521→  getStageStatuses(): Record<string, "pending" | "running" | "done" | "failed"> {
   522→    const out: Record<string, "pending" | "running" | "done" | "failed"> = {};
   523→    const stageIds = ["analyze", "plan", "architect", "generate", "build", "test", "package", "ready"];
   524→    for (const sid of stageIds) {
   525→      const s = this.stageStatus(sid);
   526→      if (s === "done") out[sid] = "done";
   527→      else if (s === "running") out[sid] = "running";
   528→      else if (s === "failed") out[sid] = "failed";
   529→      else out[sid] = "pending";
   530→    }
   531→    return out;
   532→  }
   533→
   534→  /**
   535→   * Get the real duration of a stage from its tasks' startedAt/finishedAt.
   536→   * Returns 0 if no tasks have run for the stage.
   537→   */
   538→  getStageTiming(stageId: string): number {
   539→    const tasks = this.tasksForStage(stageId);
   540→    if (tasks.length === 0) return 0;
   541→    const durations = tasks
   542→      .filter((t) => t.startedAt && t.finishedAt)
   543→      .map((t) => (t.finishedAt ?? 0) - (t.startedAt ?? 0));
   544→    return durations.length > 0 ? Math.max(...durations) : 0;
   545→  }
   546→}

   548→/* ---------------- Checkpoint Manager ---------------- */
   549→// Persists checkpoints to IndexedDB (async, large-capacity) so resume-after-
   550→// crash works for long multi-target builds. Keeps an in-memory copy for
   551→// synchronous access; IndexedDB is the durable store.
   552→export class CheckpointManager {
   553→  private checkpoints: import("./types").Checkpoint[] = [];
   554→
   555→  save(stageId: string, workflowId: WorkflowId, stageStatusSnapshot: Record<string, string>, memoryVersion: number, taskId?: string): import("./types").Checkpoint {
   556→    const cp: import("./types").Checkpoint = {
   557→      id: `cp-${Date.now()}-${Math.random().toString(36).slice(2, 6)}`,
   558→      workflowId,
   559→      stageId,
   560→      taskId,
   561→      ts: Date.now(),
   562→      stageStatusSnapshot,
   563→      memoryVersion,
   564→    };
   565→    this.checkpoints.push(cp);
   566→    // Persist to IndexedDB asynchronously (fire-and-forget; IDB is the durable
   567→    // store that survives crashes/reloads, unlike the in-memory copy).
   568→    void import("./idb").then(({ idbSaveCheckpoint }) =>
   569→      idbSaveCheckpoint({
   570→        id: cp.id,
   571→        workflowId: cp.workflowId,
   572→        stageId: cp.stageId,
   573→        taskId: cp.taskId,
   574→        ts: cp.ts,
   575→        stageStatusSnapshot: cp.stageStatusSnapshot,
   576→        memoryVersion: cp.memoryVersion,
   577→      })
   578→    );
   579→    return cp;
   580→  }
   581→
   582→  latest(): import("./types").Checkpoint | undefined {
   583→    return this.checkpoints[this.checkpoints.length - 1];
   584→  }
   585→
   586→  /** Resume from the latest checkpoint: returns the stage to resume from. */
   587→  resume(): { stageId: string; snapshot: Record<string, string> } | null {
   588→    const cp = this.latest();
   589→    if (!cp) return null;
   590→    return { stageId: cp.stageId, snapshot: cp.stageStatusSnapshot };
   591→  }
   592→
   593→  all(): import("./types").Checkpoint[] {
   594→    return [...this.checkpoints];
   595→  }
   596→
   597→  rollbackToLastGood(): import("./types").Checkpoint | undefined {
   598→    // find the last checkpoint where all stages were succeeded/pending
   599→    for (let i = this.checkpoints.length - 1; i >= 0; i--) {
   600→      const cp = this.checkpoints[i];
   601→      const allOk = Object.values(cp.stageStatusSnapshot).every(
   602→        (s) => s === "succeeded" || s === "pending"
   603→      );
   604→      if (allOk) return cp;
   605→    }
   606→    return this.checkpoints[0];
   607→  }
   608→
   609→  /**
   610→   * Restore from IndexedDB after a crash/reload. Loads the latest persisted
   611→   * checkpoint for the given workflow (or any workflow if omitted) and
   612→   * repopulates the in-memory list. Returns the stage to resume from, or null
   613→   * if nothing was persisted.
   614→   */
   615→  async restoreFromIDB(workflowId?: string): Promise<{ stageId: string; snapshot: Record<string, string> } | null> {
   616→    try {
   617→      const { idbLoadCheckpoints } = await import("./idb");
   618→      const persisted = await idbLoadCheckpoints(workflowId);
   619→      if (persisted.length === 0) return null;
   620→      // Repopulate in-memory checkpoints
   621→      this.checkpoints = persisted.map((p) => ({
   622→        id: p.id,
   623→        workflowId: p.workflowId as WorkflowId,
   624→        stageId: p.stageId,
   625→        taskId: p.taskId,
   626→        ts: p.ts,
   627→        stageStatusSnapshot: p.stageStatusSnapshot,
   628→        memoryVersion: p.memoryVersion,
   629→      }));
   630→      const latest = this.checkpoints[this.checkpoints.length - 1];
   631→      return { stageId: latest.stageId, snapshot: latest.stageStatusSnapshot };
   632→    } catch {
   633→      return null;
   634→    }
   635→  }
   636→
   637→  /** Check whether any checkpoints are persisted in IndexedDB. */
   638→  async hasPersistedState(workflowId?: string): Promise<boolean> {
   639→    try {
   640→      const { idbLoadCheckpoints } = await import("./idb");
   641→      const persisted = await idbLoadCheckpoints(workflowId);
   642→      return persisted.length > 0;
   643→    } catch {
   644→      return false;
   645→    }
   646→  }
   647→
   648→  clear(): void {
   649→    this.checkpoints = [];
   650→    // Also clear persisted state so a fresh build doesn't resume stale data.
   651→    void import("./idb").then(({ idbClearCheckpoints }) => idbClearCheckpoints());
   652→  }
   653→}
   654→
   655→export const executionEngine = new ExecutionEngine({ maxParallel: 4 });
   656→export const checkpointManager = new CheckpointManager();

   658→/* ---------------- Task factory ---------------- */
   659→let taskCounter = 0;
   660→export function makeTask(opts: {
   661→  workflowId: WorkflowId;
   662→  stageId: string;
   663→  title: string;
   664→  description: string;
   665→  agent: AgentRole;
   666→  toolId?: string;
   667→  dependsOn?: string[];
   668→  durationMs?: number;
   669→  gate?: GateId;
   670→}): Task {
   671→  return {
   672→    id: `task-${++taskCounter}`,
   673→    workflowId: opts.workflowId,
   674→    stageId: opts.stageId,
   675→    title: opts.title,
   676→    description: opts.description,
   677→    agent: opts.agent,
   678→    toolId: opts.toolId,
   679→    dependsOn: opts.dependsOn ? [...opts.dependsOn] : [],
   680→    status: "queued",
   681→    // durationMs is measured at runtime from real tool execution (startedAt→finishedAt).
   682→    // No fake random duration — default 0 until the task actually runs.
   683→    durationMs: opts.durationMs ?? 0,
   684→  };
   685→}
