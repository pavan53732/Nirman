    51‚Üí    51‚Üí * Map a target platform kind to the SKILL.md skill name that covers it.
    52‚Üí    52‚Üí * Returns undefined for kinds with no matching skill in /skills/.
    53‚Üí    53‚Üí *

    66‚Üí    66‚Üíexport function skillForTarget(kind: PlatformKind): string | undefined {

    74‚Üí    74‚Üí    case "cli":
    75‚Üí    75‚Üí      return "scaffold";
    76‚Üí    76‚Üí    case "api":
    77‚Üí    77‚Üí      return "backend-api";
    78‚Üí    78‚Üí    default:
    79‚Üí    79‚Üí      return undefined;

    83‚Üí    83‚Üí/**
    84‚Üí    84‚Üí * Client-side fetch of a SKILL.md content string from /api/skill-content.
    85‚Üí    85‚Üí * The API route reads the real file from /skills/<name>/SKILL.md via the
    86‚Üí    86‚Üí * server-only loader. Returns undefined on error or 404 (e.g. skill missing).
    87‚Üí    87‚Üí *
    88‚Üí    88‚Üí * This is how the orchestrator (which runs client-side) gets the SKILL.md
    89‚Üí    89‚Üí * content into the pipeline: the loader is server-only (uses fs), so we
    90‚Üí    90‚Üí * expose its content via an API route and fetch it here.
    91‚Üí    91‚Üí */
    92‚Üí    92‚Üíexport async function fetchSkillContent(skillName: string): Promise<string | undefined> {
    93‚Üí    93‚Üí  try {
    94‚Üí    94‚Üí    const res = await fetch(`/api/skill-content?name=${encodeURIComponent(skillName)}`);
    95‚Üí    95‚Üí    if (!res.ok) return undefined;
    96‚Üí    96‚Üí    const data = await res.json();
    97‚Üí    97‚Üí    if (!data || typeof data.content !== "string") return undefined;
    98‚Üí    98‚Üí    return data.content as string;
    99‚Üí    99‚Üí  } catch {
   100‚Üí   100‚Üí    return undefined;
   101‚Üí   101‚Üí  }
   102‚Üí   102‚Üí}

   104‚Üí   104‚Üí/**
   105‚Üí   105‚Üí * Pre-fetch SKILL.md content for every candidate target kind in parallel.
   106‚Üí   106‚Üí * Returns a partial map { kind -> content }. Kinds with no matching skill
   107‚Üí   107‚Üí * (or fetch failures) are omitted from the map. The orchestrator passes
   108‚Üí   108‚Üí * this map to detectTargets() so the Decision Engine can boost policies
   109‚Üí   109‚Üí * whose `choose` value is mentioned in the skill text.
   110‚Üí   110‚Üí */
   111‚Üí   111‚Üíexport async function fetchSkillContentForCandidateKinds(
   112‚Üí   112‚Üí  kinds: PlatformKind[]
   113‚Üí   113‚Üí): Promise<Partial<Record<PlatformKind, string>>> {
   114‚Üí   114‚Üí  const out: Partial<Record<PlatformKind, string>> = {};
   115‚Üí   115‚Üí  await Promise.all(
   116‚Üí   116‚Üí    kinds.map(async (kind) => {
   117‚Üí   117‚Üí      const skillName = skillForTarget(kind);
   118‚Üí   118‚Üí      if (!skillName) return;
   119‚Üí   119‚Üí      const content = await fetchSkillContent(skillName);
   120‚Üí   120‚Üí      if (content) out[kind] = content;
   121‚Üí   121‚Üí    })
   122‚Üí   122‚Üí  );
   123‚Üí   123‚Üí  return out;
   124‚Üí   124‚Üí}

   127‚Üí   127‚Üíexport function detectTargets(
   128‚Üí   128‚Üí  prompt: string,
   129‚Üí   129‚Üí  /**
   130‚Üí   130‚Üí   * Optional map of { platformKind -> SKILL.md content }. When supplied,
   131‚Üí   131‚Üí   * each target's pickStack() call passes the matching skill content to
   132‚Üí   132‚Üí   * decisionEngine.decide() so the skill can boost policies it endorses.
   133‚Üí   133‚Üí   * The store calls detectTargets(prompt) without this map (UI display
   134‚Üí   134‚Üí   * only ‚Äî no boost needed); the orchestrator calls it with the map after
   135‚Üí   135‚Üí   * fetching skill content from /api/skill-content.
   136‚Üí   136‚Üí   */
   137‚Üí   137‚Üí  skillContentMap?: Partial<Record<PlatformKind, string>>
   138‚Üí   138‚Üí): DetectedTargets[] {
   139‚Üí   139‚Üí  const caps = detectCapabilities(prompt);
   140‚Üí   140‚Üí  const nfs = detectNonFunctionals(prompt);
   141‚Üí   141‚Üí  const p = " " + prompt.toLowerCase() + " ";
   142‚Üí   142‚Üí  const out: DetectedTargets[] = [];
   143‚Üí   143‚Üí  const wants = (re: RegExp) => re.test(p);
   144‚Üí   144‚Üí
   145‚Üí   145‚Üí  // Domain inference: if the prompt mentions CAD/3D/GPU/game but no explicit
   146‚Üí   146‚Üí  // platform keyword, infer windows/native (not the default web). This prevents
   147‚Üí   147‚Üí  // "Build CAD software" from falling through to a generic Next.js app.
   148‚Üí   148‚Üí  const isCAD = /\b(cad|autocad|3d modeling|opengl|directx|vulkan|rendering)\b/.test(p);
   149‚Üí   149‚Üí  const isGame = /\b(game|unity|godot|2d platformer|3d game)\b/.test(p);
   150‚Üí   150‚Üí  const domainInferredWindows = isCAD && !wants(/\b(web|android|cli|api)\b/);
   151‚Üí   151‚Üí  const domainInferredGame = isGame && !wants(/\b(web|windows|android|cli|api)\b/);
   152‚Üí   152‚Üí
   153‚Üí   153‚Üí  const multi =
   154‚Üí   154‚Üí    [
   155‚Üí   155‚Üí      wants(/\b(windows|desktop|winui|wpf|winforms|win32)\b/) || domainInferredWindows,
   156‚Üí   156‚Üí      wants(/\b(android|mobile( app)?|kotlin|flutter|companion app|phone app)\b/),
   157‚Üí   157‚Üí      wants(/\b(web( site| app| admin| portal)?|website|landing|marketing site|saas|portal)\b/),
   158‚Üí   158‚Üí      wants(/\b(api|rest|backend service|microservice)\b/),
   159‚Üí   159‚Üí      wants(/\b(cli|command.line|terminal tool)\b/),
   160‚Üí   160‚Üí    ].filter(Boolean).length > 1;
   161‚Üí   161‚Üí
   162‚Üí   162‚Üí  if (wants(/\b(windows|desktop|winui|wpf|winforms|win32|\.net\s*(desktop|app))\b/) || domainInferredWindows) {
   163‚Üí   163‚Üí    const kind = "windows" as PlatformKind;
   164‚Üí   164‚Üí    const { stack, decision } = decisionEngine.pickStack(kind, prompt, caps, nfs, skillContentMap?.[kind]);
   165‚Üí   165‚Üí    out.push({ kind, label: multi ? "Desktop App" : "App", role: multi ? "Primary desktop workspace" : "Windows desktop application", stack, capabilities: caps, policies: [decision] });
   166‚Üí   166‚Üí  }
   167‚Üí   167‚Üí  if (wants(/\b(android|mobile( app)?|kotlin|flutter|companion app|phone app)\b/)) {
   168‚Üí   168‚Üí    const kind = "android" as PlatformKind;
   169‚Üí   169‚Üí    const { stack, decision } = decisionEngine.pickStack(kind, prompt, caps, nfs, skillContentMap?.[kind]);
   170‚Üí   170‚Üí    out.push({ kind, label: multi ? "Android Companion" : "App", role: multi ? "Mobile companion app" : "Android application", stack, capabilities: caps, policies: [decision] });
   171‚Üí   171‚Üí  }
   172‚Üí   172‚Üí  if (wants(/\b(web( site| app| admin| portal)?|website|landing|marketing site|saas|portal|browser)\b/)) {
   173‚Üí   173‚Üí    const kind = "web" as PlatformKind;
   174‚Üí   174‚Üí    const { stack, decision } = decisionEngine.pickStack(kind, prompt, caps, nfs, skillContentMap?.[kind]);
   175‚Üí   175‚Üí    out.push({ kind, label: multi ? "Web Portal" : "App", role: multi ? "Web admin portal" : "Web application", stack, capabilities: caps, policies: [decision] });
   176‚Üí   176‚Üí  }
   177‚Üí   177‚Üí  if (wants(/\b(api|rest( api)?|graphql|backend service|microservice)\b/)) {
   178‚Üí   178‚Üí    const kind = "api" as PlatformKind;
   179‚Üí   179‚Üí    const { stack, decision } = decisionEngine.pickStack(kind, prompt, caps, nfs, skillContentMap?.[kind]);
   180‚Üí   180‚Üí    out.push({ kind, label: "API Service", role: "Backend API and data layer", stack, capabilities: caps, policies: [decision] });
   181‚Üí   181‚Üí  }
   182‚Üí   182‚Üí  if (wants(/\b(cli|command.line|terminal tool|brew install|cargo install)\b/)) {
   183‚Üí   183‚Üí    const kind = "cli" as PlatformKind;
   184‚Üí   184‚Üí    const { stack, decision } = decisionEngine.pickStack(kind, prompt, caps, nfs, skillContentMap?.[kind]);
   185‚Üí   185‚Üí    out.push({ kind, label: "CLI Tool", role: "Command-line utility", stack, capabilities: caps, policies: [decision] });
   186‚Üí   186‚Üí  }
   187‚Üí   187‚Üí  if (wants(/\b(ai agent|autonomous agent|assistant service|chatbot|support agent)\b/)) {
   188‚Üí   188‚Üí    out.push({ kind: "api" as PlatformKind, label: "AI Agent", role: "Autonomous agent service", stack: "Python + LangGraph", capabilities: caps, policies: [decisionEngine.pickStack("api", prompt, caps, nfs, skillContentMap?.["api"]).decision] });
   189‚Üí   189‚Üí  }
   190‚Üí   190‚Üí  if (wants(/\b(library|sdk|npm package|crate)\b/)) {
   191‚Üí   191‚Üí    out.push({ kind: "library" as PlatformKind, label: "Library", role: "Reusable library / SDK", stack: /\b(rust|crate)\b/.test(p) ? "Rust crate" : "TypeScript library", capabilities: caps, policies: [] });
   192‚Üí   192‚Üí  }
   193‚Üí   193‚Üí  if (domainInferredGame) {
   194‚Üí   194‚Üí    out.push({ kind: "library" as PlatformKind, label: "Game", role: "Interactive game", stack: "Godot + GDScript", capabilities: caps, policies: [] });
   195‚Üí   195‚Üí  }
   196‚Üí   196‚Üí  if (out.length === 0) {
   197‚Üí   197‚Üí    const kind = "web" as PlatformKind;
   198‚Üí   198‚Üí    const { stack, decision } = decisionEngine.pickStack(kind, prompt, caps, nfs, skillContentMap?.[kind]);
   199‚Üí   199‚Üí    out.push({ kind, label: "Web App", role: "Web application", stack, capabilities: caps, policies: [decision] });
   200‚Üí   200‚Üí  }
   201‚Üí   201‚Üí  return out;
   202‚Üí   202‚Üí}

   204‚Üí   204‚Üíexport class Orchestrator {
   205‚Üí   205‚Üí  /** Bootstrap: register all data into registries and wire the event bus. */
   206‚Üí   206‚Üí  bootstrap(): void {
   207‚Üí   207‚Üí    // registries are populated by the index module import side-effects;
   208‚Üí   208‚Üí    // here we just wire observability to the execution engine event bus.
   209‚Üí   209‚Üí    executionEngine.subscribe((e) => {
   210‚Üí   210‚Üí      observability.recordEvent(e);
   211‚Üí   211‚Üí    });
   212‚Üí   212‚Üí  }
   213‚Üí   213‚Üí
   214‚Üí   214‚Üí  /**
   215‚Üí   215‚Üí   * Run a build: select workflow, detect targets + capabilities, write
   216‚Üí   216‚Üí   * requirements/decision memory, compile the workflow DAG, submit to the
   217‚Üí   217‚Üí   * execution engine, and return the plan.
   218‚Üí   218‚Üí   */
   219‚Üí   219‚Üí  async startBuild(prompt: string, projectId?: string): Promise<OrchestrationResult & { tasks: Task[] }> {
   220‚Üí   220‚Üí    executionEngine.reset();
   221‚Üí   221‚Üí    checkpointManager.clear();
   222‚Üí   222‚Üí    artifactRegistry.clear();
   223‚Üí   223‚Üí
   224‚Üí   224‚Üí    // Context Builder pulls relevant memory for the planner
   225‚Üí   225‚Üí    const ctx = contextBuilder.buildForAgent("planner", { kinds: ["requirements", "decision", "conversation"], prompt });
   226‚Üí   226‚Üí
   227‚Üí   227‚Üí    // Capability Detection (runs first)
   228‚Üí   228‚Üí    const capabilities = detectCapabilities(prompt);
   229‚Üí   229‚Üí    observability.recordEvent({
   230‚Üí   230‚Üí      id: `ev-${Date.now()}`,
   231‚Üí   231‚Üí      ts: Date.now(),
   232‚Üí   232‚Üí      type: "capability-detected",
   233‚Üí   233‚Üí      message: `Detected capabilities: ${capabilities.join(", ") || "none"}`,
   234‚Üí   234‚Üí      level: "info",
   235‚Üí   235‚Üí    });
   236‚Üí   236‚Üí
   237‚Üí   237‚Üí    // Autonomy gate: Ambiguity Detection. If the requirement is too ambiguous
   238‚Üí   238‚Üí    // (score > AMBIGUITY_THRESHOLD), emit a human-question and pause the
   239‚Üí   239‚Üí    // workflow instead of inventing business requirements. The engine asks
   240‚Üí   240‚Üí    // ONLY when information is missing, conflicting, or an external resource
   241‚Üí   241‚Üí    // is needed without credentials.
   242‚Üí   242‚Üí    const ambiguity = detectAmbiguity(prompt);
   243‚Üí   243‚Üí    observability.recordEvent({
   244‚Üí   244‚Üí      id: `ev-${Date.now()}-amb`,
   245‚Üí   245‚Üí      ts: Date.now(),
   246‚Üí   246‚Üí      type: "capability-detected",
   247‚Üí   247‚Üí      message: `Ambiguity score ${ambiguity.score.toFixed(2)} (threshold ${AMBIGUITY_THRESHOLD})${ambiguity.shouldAsk ? " ‚Üí asking user" : " ‚Üí proceeding autonomously"}`,
   248‚Üí   248‚Üí      level: ambiguity.shouldAsk ? "warn" : "info",
   249‚Üí   249‚Üí    });
   250‚Üí   250‚Üí    const pendingQuestion = askQuestionIfNeeded(prompt);
   251‚Üí   251‚Üí
   252‚Üí   252‚Üí    // ---- Phase A1: Pre-fetch SKILL.md content for every candidate target ----
   253‚Üí   253‚Üí    // The orchestrator runs client-side, so it fetches the skill content via
   254‚Üí   254‚Üí    // /api/skill-content (the server-only loader reads /skills/<name>/SKILL.md).
   255‚Üí   255‚Üí    // The map is passed to detectTargets() so the Decision Engine can boost
   256‚Üí   256‚Üí    // policies whose `choose` value is mentioned in the skill text, and to
   257‚Üí   257‚Üí    // generateForTarget() so the generators can include the skill guidance as
   258‚Üí   258‚Üí    // a comment header in their README.
   259‚Üí   259‚Üí    const candidateKinds: PlatformKind[] = ["windows", "android", "web", "api", "cli"];
   260‚Üí   260‚Üí    const skillContentMap = await fetchSkillContentForCandidateKinds(candidateKinds);
   261‚Üí   261‚Üí    const skillFetchCount = Object.keys(skillContentMap).length;
   262‚Üí   262‚Üí    if (skillFetchCount > 0) {
   263‚Üí   263‚Üí      observability.recordEvent({
   264‚Üí   264‚Üí        id: `ev-${Date.now()}-skill`,
   265‚Üí   265‚Üí        ts: Date.now(),
   266‚Üí   266‚Üí        type: "memory-written",
   267‚Üí   267‚Üí        message: `Loaded ${skillFetchCount} SKILL.md file(s) from /api/skill-content: ${candidateKinds
   268‚Üí   268‚Üí          .filter((k) => skillContentMap[k])
   269‚Üí   269‚Üí          .map((k) => `${k}‚Üí${skillForTarget(k)}`)
   270‚Üí   270‚Üí          .join(", ")}`,
   271‚Üí   271‚Üí        level: "info",
   272‚Üí   272‚Üí      });
   273‚Üí   273‚Üí    }
   274‚Üí   274‚Üí
   275‚Üí   275‚Üí    // Detect multi-targets + decisions (with skill-content boost)
   276‚Üí   276‚Üí    const targets = detectTargets(prompt, skillContentMap);
   277‚Üí   277‚Üí    const decisions = targets.flatMap((t) => t.policies);
   278‚Üí   278‚Üí
   279‚Üí   279‚Üí    // Write Requirements Memory + Decision Memory
   280‚Üí   280‚Üí    projectMemory.write("requirements", "Original Prompt", prompt, "user");
   281‚Üí   281‚Üí    projectMemory.write("requirements", "Detected Targets", targets.map((t) => `${t.label}: ${t.stack} (${t.role})`).join("\n"), "decision-engine");

   559‚Üí   559‚Üí/** Derive a clean project name from the prompt (shared with the store). */
   560‚Üí   560‚Üífunction promptToName(prompt: string): string {
   561‚Üí   561‚Üí  let trimmed = prompt.trim().toLowerCase();
   562‚Üí   562‚Üí  let prev = "";
   563‚Üí   563‚Üí  while (prev !== trimmed) {
   564‚Üí   564‚Üí    prev = trimmed;
   565‚Üí   565‚Üí    trimmed = trimmed.replace(
   566‚Üí   566‚Üí      /^(build|build me|create|make|generate|develop|i want|i need|please|a|an|the|me|some)\s+/i,
   567‚Üí   567‚Üí      ""
   568‚Üí   568‚Üí    );
   569‚Üí   569‚Üí  }
