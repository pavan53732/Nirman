    44→// Debounced trace sync — POSTs the client-side executionEngine trace + agent
    45→// activations to the server so /api/build/trace and /api/agents/trace return
    46→// real data after a client-side build. The orchestrator runs in the browser
    47→// (Zustand store), so without this sync the server-side trace endpoints would
    48→// always return empty.
    49→let syncTimer: ReturnType<typeof setTimeout> | null = null;
    50→function scheduleTraceSync(): void {
    51→  if (typeof window === "undefined") return; // SSR guard
    52→  if (syncTimer) clearTimeout(syncTimer);
    53→  syncTimer = setTimeout(async () => {
    54→    syncTimer = null;
    55→    try {
    56→      const trace = executionEngine.getTrace();
    57→      await fetch("/api/build/trace", {
    58→        method: "POST",
    59→        headers: { "Content-Type": "application/json" },
    60→        body: JSON.stringify({ trace }),
    61→      });
    62→    } catch { /* best-effort */ }
    63→    try {
    64→      const activations = agentRuntime.getActivations();
    65→      const summary = agentRuntime.getSummary();
    66→      await fetch("/api/agents/trace", {
    67→        method: "POST",
    68→        headers: { "Content-Type": "application/json" },
    69→        body: JSON.stringify({ summary, activations }),
    70→      });
    71→    } catch { /* best-effort */ }
    72→  }, 250);
    73→}

    87→/** Detect generation targets (multi-target) from a prompt. */
    88→export function detectTargets(prompt: string): DetectedTargets[] {
    89→  const caps = detectCapabilities(prompt);
    90→  const nfs = detectNonFunctionals(prompt);
    91→  const p = " " + prompt.toLowerCase() + " ";
    92→  const out: DetectedTargets[] = [];
    93→  const wants = (re: RegExp) => re.test(p);
    94→
    95→  // Domain inference: if the prompt mentions CAD/3D/GPU/game but no explicit
    96→  // platform keyword, infer windows/native (not the default web). This prevents
    97→  // "Build CAD software" from falling through to a generic Next.js app.
    98→  const isCAD = /\b(cad|autocad|3d modeling|opengl|directx|vulkan|rendering)\b/.test(p);
    99→  const isGame = /\b(game|unity|godot|2d platformer|3d game)\b/.test(p);
   100→  const domainInferredWindows = isCAD && !wants(/\b(web|android|cli|api)\b/);
   101→  const domainInferredGame = isGame && !wants(/\b(web|windows|android|cli|api)\b/);
   102→
   103→  const multi =
   104→    [
   105→      wants(/\b(windows|desktop|winui|wpf|winforms|win32)\b/) || domainInferredWindows,
   106→      wants(/\b(android|mobile( app)?|kotlin|flutter|companion app|phone app)\b/),
   107→      wants(/\b(web( site| app| admin| portal)?|website|landing|marketing site|saas|portal)\b/),
   108→      wants(/\b(api|rest|backend service|microservice)\b/),
   109→      wants(/\b(cli|command.line|terminal tool)\b/),
   110→    ].filter(Boolean).length > 1;
   111→
   112→  if (wants(/\b(windows|desktop|winui|wpf|winforms|win32|\.net\s*(desktop|app))\b/) || domainInferredWindows) {
   113→    const kind = "windows" as PlatformKind;
   114→    const { stack, decision } = decisionEngine.pickStack(kind, prompt, caps, nfs);
   115→    out.push({ kind, label: multi ? "Desktop App" : "App", role: multi ? "Primary desktop workspace" : "Windows desktop application", stack, capabilities: caps, policies: [decision] });
   116→  }
   117→  if (wants(/\b(android|mobile( app)?|kotlin|flutter|companion app|phone app)\b/)) {
   118→    const kind = "android" as PlatformKind;
   119→    const { stack, decision } = decisionEngine.pickStack(kind, prompt, caps, nfs);
   120→    out.push({ kind, label: multi ? "Android Companion" : "App", role: multi ? "Mobile companion app" : "Android application", stack, capabilities: caps, policies: [decision] });
   121→  }
   122→  if (wants(/\b(web( site| app| admin| portal)?|website|landing|marketing site|saas|portal|browser)\b/)) {
   123→    const kind = "web" as PlatformKind;
   124→    const { stack, decision } = decisionEngine.pickStack(kind, prompt, caps, nfs);
   125→    out.push({ kind, label: multi ? "Web Portal" : "App", role: multi ? "Web admin portal" : "Web application", stack, capabilities: caps, policies: [decision] });
   126→  }
   127→  if (wants(/\b(api|rest( api)?|graphql|backend service|microservice)\b/)) {
   128→    const kind = "api" as PlatformKind;
   129→    const { stack, decision } = decisionEngine.pickStack(kind, prompt, caps, nfs);
   130→    out.push({ kind, label: "API Service", role: "Backend API and data layer", stack, capabilities: caps, policies: [decision] });
   131→  }
   132→  if (wants(/\b(cli|command.line|terminal tool|brew install|cargo install)\b/)) {
   133→    const kind = "cli" as PlatformKind;
   134→    const { stack, decision } = decisionEngine.pickStack(kind, prompt, caps, nfs);
   135→    out.push({ kind, label: "CLI Tool", role: "Command-line utility", stack, capabilities: caps, policies: [decision] });
   136→  }
   137→  if (wants(/\b(ai agent|autonomous agent|assistant service|chatbot|support agent)\b/)) {
   138→    out.push({ kind: "api" as PlatformKind, label: "AI Agent", role: "Autonomous agent service", stack: "Python + LangGraph", capabilities: caps, policies: [decisionEngine.pickStack("api", prompt, caps, nfs).decision] });
   139→  }
   140→  if (wants(/\b(library|sdk|npm package|crate)\b/)) {
   141→    out.push({ kind: "library" as PlatformKind, label: "Library", role: "Reusable library / SDK", stack: /\b(rust|crate)\b/.test(p) ? "Rust crate" : "TypeScript library", capabilities: caps, policies: [] });
   142→  }
   143→  if (domainInferredGame) {
   144→    out.push({ kind: "library" as PlatformKind, label: "Game", role: "Interactive game", stack: "Godot + GDScript", capabilities: caps, policies: [] });
   145→  }
   146→  if (out.length === 0) {
   147→    const kind = "web" as PlatformKind;
   148→    const { stack, decision } = decisionEngine.pickStack(kind, prompt, caps, nfs);
   149→    out.push({ kind, label: "Web App", role: "Web application", stack, capabilities: caps, policies: [decision] });
   150→  }
   151→  return out;
   152→}

   154→export class Orchestrator {
   155→  /** Bootstrap: register all data into registries and wire the event bus. */
   156→  bootstrap(): void {
   157→    // registries are populated by the index module import side-effects;
   158→    // here we just wire observability to the execution engine event bus.
   159→    initAgentRuntime();
   160→    executionEngine.subscribe((e) => {
   161→      observability.recordEvent(e);
   162→      // Sync trace + agent activations to the server so the debug endpoints
   163→      // return real data after a client-side build. Debounced to avoid
   164→      // flooding the server with one POST per event.
   165→      scheduleTraceSync();
   166→    });
   167→  }
   168→
   169→  /**
   170→   * Run a build: select workflow, detect targets + capabilities, write
   171→   * requirements/decision memory, compile the workflow DAG, submit to the
   172→   * execution engine, and return the plan.
   173→   */
   174→  async startBuild(prompt: string, projectId?: string): Promise<OrchestrationResult & { tasks: Task[] }> {
   175→    executionEngine.reset();
   176→    checkpointManager.clear();
   177→    artifactRegistry.clear();
   178→
   179→    // Context Builder pulls relevant memory for the planner
   180→    const ctx = contextBuilder.buildForAgent("planner", { kinds: ["requirements", "decision", "conversation"], prompt });
   181→
   182→    // Capability Detection (runs first)
   183→    const capabilities = detectCapabilities(prompt);
   184→    observability.recordEvent({
   185→      id: `ev-${Date.now()}`,
   186→      ts: Date.now(),
   187→      type: "capability-detected",
   188→      message: `Detected capabilities: ${capabilities.join(", ") || "none"}`,
   189→      level: "info",
   190→    });
   191→
   192→    // Autonomy gate: Ambiguity Detection. If the requirement is too ambiguous
   193→    // (score > AMBIGUITY_THRESHOLD), emit a human-question and pause the
   194→    // workflow instead of inventing business requirements. The engine asks
   195→    // ONLY when information is missing, conflicting, or an external resource
   196→    // is needed without credentials.
   197→    const ambiguity = detectAmbiguity(prompt);
   198→    observability.recordEvent({
   199→      id: `ev-${Date.now()}-amb`,
   200→      ts: Date.now(),
   201→      type: "capability-detected",
   202→      message: `Ambiguity score ${ambiguity.score.toFixed(2)} (threshold ${AMBIGUITY_THRESHOLD})${ambiguity.shouldAsk ? " → asking user" : " → proceeding autonomously"}`,
   203→      level: ambiguity.shouldAsk ? "warn" : "info",
   204→    });
   205→    const pendingQuestion = askQuestionIfNeeded(prompt);
   206→
   207→    // Detect multi-targets + decisions
   208→    const targets = detectTargets(prompt);
   209→    const decisions = targets.flatMap((t) => t.policies);
   210→
   211→    // Write Requirements Memory + Decision Memory
   212→    projectMemory.write("requirements", "Original Prompt", prompt, "user");
   213→    projectMemory.write("requirements", "Detected Targets", targets.map((t) => `${t.label}: ${t.stack} (${t.role})`).join("\n"), "decision-engine");
   214→    projectMemory.write("decision", "Stack Selection", decisions.map((d) => `${d.topic}: ${d.chosen} (${d.confidence})`).join("\n"), "decision-engine");
   215→    projectMemory.write("architecture", "Capabilities", capabilities.join(", ") || "none", "decision-engine");
   216→
   217→    // ---- READ Architecture Memory: database choice drives generator output ----
   218→    // The user (or a future LLM agent) may write "Database: PostgreSQL" to
   219→    // Architecture Memory. Generators branch on this — Prisma provider, EF Core
   220→    // provider, Android persistence note. Reading memory HERE proves memory has
   221→    // real impact on generated output (see /api/debug/memory-impact).
   222→    const database = readDatabaseFromMemory();
   223→    if (database !== "sqlite") {
   224→      // Surface the parsed choice in Decision Memory so the UI + downstream
   225→      // agents can see that the database was overridden by Architecture Memory.
   226→      projectMemory.write(
   227→        "decision",
   228→        "Database Choice (from Architecture Memory)",
   229→        database,
   230→        "orchestrator"
   231→      );
   232→    }
   233→
   234→    // ---- SharedContext: fresh blackboard + initial context writes ----------
   235→    // Clear any stale entries from a previous build, then publish the key
   236→    // context every downstream agent will read. Agents no longer receive
   237→    // this via direct calls — they read it from the shared blackboard:
   238→    //   Planner   reads "prompt" + "capabilities"
   239→    //   Architect reads "plan" + "targets" + "decisions"
   240→    //   Generator reads "architecture" + "targets[i]"
   241→    //   Reviewer  reads "code:<target>"
   242→    //   Builder   reads "review:<target>"
   243→    // Keys written here are the build's initial inputs; per-agent outputs
   244→    // are written by the agents themselves (Task I handlers).
   245→    sharedContext.clear();
   246→    sharedContext.write("prompt", prompt);
   247→    sharedContext.write("targets", targets);
   248→    sharedContext.write("capabilities", capabilities);
   249→    sharedContext.write("decisions", decisions);
   250→    sharedContext.write("database", database);
   251→    sharedContext.write("projectId", projectId ?? `proj-${Date.now()}`);
   252→    observability.recordEvent({
   253→      id: `ev-${Date.now()}-shared`,
   254→      ts: Date.now(),
   255→      type: "memory-written",
   256→      message: `SharedContext initialized: ${sharedContext.readAll ? Object.keys(sharedContext.readAll()).length : 0} keys (prompt, targets, capabilities, decisions, database, projectId)`,
   257→      level: "debug",
   258→    });
   259→
   260→    // Workflow selection
   261→    const workflow = workflowEngine.select(prompt);
   262→    observability.recordEvent({
   263→      id: `ev-${Date.now()}-wf`,
   264→      ts: Date.now(),
   265→      type: "workflow-selected",
   266→      workflowId: workflow.id,
   267→      message: `Selected workflow: ${workflow.name}`,
   268→      level: "info",
   269→    });
   270→
   271→    // ---- Real generation: invoke the generator for each detected target ----
   272→    // The Desktop Generator (Anvil) produces WinUI 3 / Tauri scaffolding; the
   273→    // Android Generator (Droid) produces Kotlin+Compose / Flutter; the Web
   274→    // Generator (Forge) produces a REAL compilable Next.js app (Prisma + auth
   275→    // + CRUD pages derived from the requirement's data model). Files are
   276→    // versioned into the Artifact Registry and emitted as artifact-produced
   277→    // events for observability.
   278→    const generationResults = targets.map((t, i) => {
   279→      const result = generateForTarget(t.kind, t.stack, promptToName(prompt) || `App${i + 1}`, `t${i + 1}`, {
   280→        prompt,
   281→        capabilities,
   282→        nonFunctionals: detectNonFunctionals(prompt),
   283→        database,
   284→      });
   285→      observability.recordEvent({
   286→        id: `ev-${Date.now()}-gen-${i}`,
   287→        ts: Date.now(),
   288→        type: "artifact-produced",
   289→        stageId: "generate",
   290→        message: `${t.label}: generated ${result.files.length} files (${result.stack})`,
   291→        level: "success",
   292→      });
   293→      projectMemory.write(
   294→        "code",
   295→        `${t.label} source`,
   296→        result.files.map((f) => `${f.path} (${(f.content.length / 1024).toFixed(1)} KB)`).join("\n"),
   297→        result.producedBy
   298→      );
   299→      return result;
   300→    });
   301→    const totalFiles = generationResults.reduce((n, g) => n + g.files.length, 0);
   302→
   303→    // Publish per-target generation outputs to the SharedContext blackboard
   304→    // (key convention: "code:<target>"). Downstream agents (Reviewer, Builder)
   305→    // read these instead of receiving them via direct calls. The direct
   306→    // generation above is the backward-compat fallback; when Task I's
   307→    // handlers are wired, the per-target generation tasks below will produce
   308→    // these same keys via AgentExecutionResult.sharedWrites.
   309→    for (let i = 0; i < targets.length; i++) {
   310→      const t = targets[i];
   311→      const g = generationResults[i];
   312→      sharedContext.write(`code:${t.kind}`, g.files);
   313→    }
   314→    sharedContext.write("generatedFilesCount", totalFiles);
   315→
   316→    // Compile DAG
   317→    const tasks = workflowEngine.compile(workflow, workflow.stages.map((s) => s.id as StageId));
   318→
   319→    // ---- Parallel per-target generation tasks ------------------------------
   320→    // The compiled DAG has a SINGLE "generate" stage task. We augment it
   321→    // with one "Generating (<target>)" task per detected target. These
   322→    // tasks all depend on the SAME architecture predecessors (the architect
   323→    // stage task + architect gate) and have NO dependencies on each other →
   324→    // the ExecutionEngine dispatches them in the SAME parallelBatch
   325→    // (maxParallel=4), proving true parallel scheduling in /api/build/trace.
   326→    //
   327→    // Agent attribution uses the kind-appropriate generator so the agent
   328→    // trace shows the right specialist per target (Forge=web, Anvil=desktop,
   329→    // Droid=android, etc.).
   330→    //
   331→    // Each task carries the target spec in its description (JSON-encoded)
   332→    // so the AgentRuntime handler (Task I) can read it and call
   333→    // generateForTarget with the right arguments. Until Task I's executor
   334→    // is wired, these tasks complete as no-ops — the real generation
   335→    // already ran above (backward-compat fallback) and the workspace is
   336→    // already materialized below.
   337→    const generateStageTask = tasks.find((t) => t.stageId === "generate" && !t.gate);
   338→    // The generate stage task's dependsOn are the architect stage task +
   339→    // architect gate — i.e. the architecture predecessors. The per-target
   340→    // gen tasks share these so they become ready in the SAME scheduler
   341→    // tick as the generate stage task → same parallelBatch.
   342→    let archDeps: string[] = generateStageTask?.dependsOn ?? [];
   343→    if (archDeps.length === 0) {
   344→      // Fallback for workflows without an explicit generate stage task:
   345→      // depend on the architect gate (or architect stage task) directly.
   346→      const archGate = tasks.find((t) => t.stageId === "architect" && t.gate === "architecture");
   347→      const archStage = tasks.find((t) => t.stageId === "architect" && !t.gate);
   348→      archDeps = archGate?.id ? [archGate.id] : archStage?.id ? [archStage.id] : [];
   349→    }
   350→
   351→    // Map each platform kind to its specialist generator agent.
   352→    const targetAgent: Partial<Record<PlatformKind, AgentRole>> = {
   353→      windows: "desktop-generator",
   354→      android: "android-generator",
   355→      web: "frontend-generator",
   356→      api: "backend-generator",
   357→      cli: "backend-generator",
   358→      library: "backend-generator",
   359→    };
   360→
   361→    const perTargetGenTasks: Task[] = targets.map((t, i) => {
   362→      const targetId = `t${i + 1}`;
   363→      const name = promptToName(prompt) || `App${i + 1}`;
   364→      const task = makeTask({
   365→        workflowId: workflow.id,
   366→        stageId: "generate",
   367→        title: `Generating (${t.label})`,
   368→        description: JSON.stringify({
   369→          kind: t.kind,
   370→          stack: t.stack,
   371→          name,
   372→          targetId,
   373→          label: t.label,
   374→          role: t.role,
   375→          capabilities: t.capabilities,
   376→          database,
   377→        }),
   378→        agent: targetAgent[t.kind] ?? "frontend-generator",
   379→        dependsOn: [...archDeps], // copy — same as generate stage task → parallel
   380→      });
   381→      return task;
   382→    });
   383→
   384→    // Insert the per-target tasks right after the generate stage task so the
   385→    // trace's scheduledAt ordering is stable + readable. If no generate stage
   386→    // task exists, append them at the end.
   387→    if (generateStageTask) {
   388→      const idx = tasks.indexOf(generateStageTask);
   389→      tasks.splice(idx + 1, 0, ...perTargetGenTasks);
   390→    } else {
   391→      tasks.push(...perTargetGenTasks);
   392→    }
   393→
   394→    // Extend the generate stage's compilation gate to also depend on the
   395→    // per-target tasks — otherwise the gate would fire as soon as the
   396→    // (no-op) generate stage task completes, racing the parallel tasks.
   397→    // Now the gate waits for ALL parallel generation tasks to finish.
   398→    for (const t of tasks) {
   399→      if (t.stageId === "generate" && t.gate === "compilation") {
   400→        for (const pt of perTargetGenTasks) {
   401→          if (!t.dependsOn.includes(pt.id)) t.dependsOn.push(pt.id);
   402→        }
   403→      }
   404→    }
   405→
   406→    observability.recordEvent({
   407→      id: `ev-${Date.now()}-par-gen`,
   408→      ts: Date.now(),
   409→      type: "task-queued",
   410→      stageId: "generate",
   411→      message: `Created ${perTargetGenTasks.length} parallel generation tasks (dependsOn arch: [${archDeps.join(", ")}])`,
   412→      level: "info",
   413→    });
   414→
   415→    // Materialize generated files to a real on-disk workspace so the
   416→    // compilation gate can run `tsc --noEmit` against them. Attach the
   417→    // workspace path to the compilation gate task + the build stage task.
   418→    const wsProjectId = projectId ?? `proj-${Date.now()}`;
   419→    const workspacePaths: Record<string, string> = {}; // targetId -> path
   420→    for (let i = 0; i < targets.length; i++) {
   421→      const t = targets[i];
   422→      const gen = generationResults[i];
   423→      const folder = t.kind === "windows" ? "desktop" : t.kind === "android" ? "android" : t.kind === "web" ? "web-admin" : t.kind === "cli" ? "cli" : "app";
   424→      try {
   425→        const res = await fetch("/api/workspace", {
   426→          method: "POST",
   427→          headers: { "Content-Type": "application/json" },
   428→          body: JSON.stringify({ projectId: wsProjectId, targetFolder: folder, files: gen.files }),
   429→        });
   430→        if (res.ok) {
   431→          const { path: wsPath } = await res.json();
   432→          workspacePaths[`t${i + 1}`] = wsPath;
   433→          observability.recordEvent({
   434→            id: `ev-${Date.now()}-ws-${i}`,
   435→            ts: Date.now(),
   436→            type: "artifact-produced",
   437→            stageId: "generate",
   438→            message: `${t.label}: materialized ${gen.files.length} files to ${wsPath}`,
   439→            level: "info",
   440→          });
   441→        }
   442→      } catch {
   443→        // workspace write is best-effort; gates will skip if no workspace
   444→      }
   445→    }
   446→    // Publish per-target workspace paths to the SharedContext blackboard.
   447→    // Downstream build/review agents read "workspace:<target>" to find the
   448→    // on-disk location of each target's materialized files.
   449→    sharedContext.write("workspaces", workspacePaths);
   450→    for (let i = 0; i < targets.length; i++) {
   451→      const ws = workspacePaths[`t${i + 1}`];
   452→      if (ws) sharedContext.write(`workspace:${targets[i].kind}`, ws);
   453→    }
   454→
   455→    // Annotate the compilation gate task with the web workspace (runs tsc).
   456→    // For multi-target builds, attach the primary target's workspace + type.
   457→    // Desktop/android static validation happens via their own gate tasks below.
   458→    const webWorkspace = workspacePaths["t1"] ?? Object.values(workspacePaths)[0] ?? undefined;
   459→    const primaryTarget = targets[0];
   460→    const primaryType = primaryTarget?.kind === "windows" ? "desktop" : primaryTarget?.kind === "android" ? "android" : "web";
   461→    for (const task of tasks) {
   462→      // Set gateContext for ALL gate tasks. Structural gates (architecture,
   463→      // security, etc.) need artifactCount to pass. Compilation gates need
   464→      // workspacePath + targetType.
   465→      if (task.gate) {
   466→        (task as Task & { gateContext?: import("./self-healing").GateEvaluationContext }).gateContext = {
   467→          workspacePath: webWorkspace,
   468→          artifactCount: totalFiles,
   469→          targetType: task.gate === "compilation" ? primaryType : undefined,
   470→        };
   471→      }
   472→      // Attach toolId + cwd to the build stage task so it runs npm-build (web only)
   473→      if (task.stageId === "build" && !task.gate && webWorkspace && primaryTarget?.kind === "web") {
   474→        task.toolId = "npm-build";
   475→        (task as Task & { args?: { cwd?: string } }).args = { cwd: webWorkspace };
   476→      }
   477→    }
   478→
   479→    // For multi-target builds, add per-target compilation gate tasks for
   480→    // desktop/android so their static validators run too.
   481→    for (let i = 0; i < targets.length; i++) {
   482→      const t = targets[i];
   483→      const wsPath = workspacePaths[`t${i + 1}`];
   484→      if (!wsPath) continue;
   485→      const tType = t.kind === "windows" ? "desktop" : t.kind === "android" ? "android" : null;
   486→      if (!tType || tType === primaryType) continue; // skip, already covered
   487→      const extraGate = makeTask({
   488→        workflowId: workflow.id,
   489→        stageId: "build",
   490→        title: `Gate: compilation (${tType})`,
   491→        description: `Static validation for ${t.label}`,
   492→        agent: "orchestrator" as AgentRole,
   493→        dependsOn: [],
   494→        gate: "compilation" as GateId,
   495→      });
   496→      (extraGate as Task & { gateContext?: import("./self-healing").GateEvaluationContext }).gateContext = {
   497→        workspacePath: wsPath,
   498→        artifactCount: totalFiles,
   499→        targetType: tType as "desktop" | "android",
   500→      };
   501→      tasks.push(extraGate);
   502→    }
   503→
   504→    // Token budget check
   505→    if (!tokenBudgetManager.withinBudget("planner", workflow.id)) {
   506→      observability.recordEvent({
   507→        id: `ev-${Date.now()}-budget`,
   508→        ts: Date.now(),
   509→        type: "task-failed",
   510→        message: "Token budget exhausted for planner",
   511→        level: "error",
   512→      });
   513→    }
   514→    // Token charging: real tokens come from the z-ai SDK chat response
   515→    // (captured in use-chat.ts via usage.total_tokens). For non-LLM tasks
   516→    // (planner context, generator file production), tokens = 0 — no estimate.
   517→    // The ctx.tokenEstimate is NOT charged — it was only used for budget
   518→    // checking, not for observability metrics.
   519→    tokenBudgetManager.charge("planner", workflow.id, 0);
   520→    // Generator agents produce files via deterministic templates, not LLM
   521→    // calls — no tokens consumed. Real tokens are only from /api/chat.
   522→    for (const g of generationResults) {
   523→      observability.chargeTokens(g.producedBy, 0, workflow.id);
   524→      tokenBudgetManager.charge(g.producedBy, workflow.id, 0);
   525→    }
   526→
   527→    // Submit to execution engine (parallel, dependency-scheduled). If the
   528→    // ambiguity gate raised a question, askQuestionIfNeeded already cancelled
   529→    // running tasks; we still return the plan so the UI can surface the question.
   530→    executionEngine.submitAll(tasks);
   531→    if (pendingQuestion) {
   532→      // Pause: cancel anything that just started so the engine waits for the
   533→      // user's clarification before resuming.
   534→      executionEngine.cancelAll();
   535→      projectMemory.write("requirements", "Pending Question", pendingQuestion, "ambiguity-detector");
   536→    }
   537→
   538→    return {
   539→      workflow,
   540→      targets,
   541→      decisions,
   542→      capabilities,
   543→      tasks,
   544→      generatedFiles: totalFiles,
   545→      ambiguityScore: ambiguity.score,
   546→      pendingQuestion,
   547→    };
   548→  }

   591→/**
   592→ * Read the configured database choice from Architecture Memory.
   593→ *
   594→ * Generators must branch on the database selected for the project (e.g.
   595→ * SQLite for offline-first vs PostgreSQL for client/server). Architecture
   596→ * Memory is the source of truth — when an LLM agent (or the user) writes
   597→ * "Database: PostgreSQL" there, every subsequent generator run should emit
   598→ * the PostgreSQL flavor of the Prisma schema / EF Core provider / Android
   599→ * note. This helper reads memory and parses the choice.
   600→ *
   601→ * Order matters: PostgreSQL is checked first so a record that says
   602→ * "Database: PostgreSQL" wins even if older records mentioned SQLite.
   603→ * Defaults to "sqlite" when memory is empty or no database record exists.
   604→ *
   605→ * Records are matched on lowercased `content`; the title is ignored so the
   606→ * helper works with any naming convention (e.g. "Database", "Database Choice",
   607→ * "Storage", etc.). Both `database: postgres` and bare `postgresql` match.
   608→ */
   609→export function readDatabaseFromMemory(): DatabaseChoice {
   610→  const records = projectMemory.read("architecture");
   611→  // Scan newest-first so the latest database decision wins.
   612→  for (const r of [...records].sort((a, b) => b.createdAt - a.createdAt)) {
   613→    const content = r.content.toLowerCase();
   614→    if (/database\s*[:=]?\s*postgres/.test(content) || /\bpostgresql\b/.test(content)) {
   615→      return "postgresql";
   616→    }
   617→    if (/database\s*[:=]?\s*sqlite/.test(content) || /\bsqlite\b/.test(content)) {
   618→      return "sqlite";
   619→    }
   620→  }
   621→  return "sqlite"; // default — matches the existing offline-first behavior
   622→}

   624→/** Derive a clean project name from the prompt (shared with the store). */
   625→function promptToName(prompt: string): string {
   626→  let trimmed = prompt.trim().toLowerCase();
   627→  let prev = "";
   628→  while (prev !== trimmed) {
   629→    prev = trimmed;
   630→    trimmed = trimmed.replace(
   631→      /^(build|build me|create|make|generate|develop|i want|i need|please|a|an|the|me|some)\s+/i,
   632→      ""
   633→    );
   634→  }
   635→  const stopwords = new Set([
   636→    "in", "that", "with", "for", "to", "and", "of", "on", "using", "via",
   637→    "which", "from", "into", "where", "when", "app", "application", "as",
   638→    "companion", "portal", "admin", "me",
   639→  ]);
   640→  const acronyms = new Set(["cli", "ai", "api", "sdk", "saas", "ui", "ux", "ios", "ml", "crm", "cms", "erp", "hrm"]);
   641→  const words: string[] = [];
   642→  for (const w of trimmed.split(/\s+/).filter(Boolean)) {
   643→    if (stopwords.has(w)) break;
   644→    words.push(w);
   645→    if (words.length >= 3) break;
   646→  }
   647→  if (words.length === 0) return "App";
   648→  return words
   649→    .map((w) => (acronyms.has(w) ? w.toUpperCase() : w.charAt(0).toUpperCase() + w.slice(1)))
   650→    .join(" ")
   651→    .replace(/[^a-zA-Z0-9 ]/g, "")
   652→    .trim();
   653→}
