    53→/**
    54→ * Map a target platform kind to the SKILL.md skill name that covers it.
    55→ * Returns undefined for kinds with no matching skill in /skills/.
    56→ *
    57→ *   windows → windows-native  (WinUI 3 / WPF / .NET MAUI / Win32 guidance)
    58→ *   web     → fullstack-app   (full-stack web app orchestration)
    59→ *   android → frontend-setup  (closest match: UI/frontend guidance applies
    60→ *                              to Compose UI as well; no android-specific
    61→ *                              skill exists yet)
    62→ *   cli     → scaffold        (project scaffolding guidance)
    63→ *   api     → backend-api     (backend API design)
    64→ *
    65→ * "desktop-app" is the desktop fallback skill (cross-platform desktop
    66→ * framework choice) — used when a windows target is detected but the user
    67→ * didn't explicitly ask for native.
    68→ */
    69→export function skillForTarget(kind: PlatformKind): string | undefined {
    70→  switch (kind) {
    71→    case "windows":
    72→      return "windows-native";
    73→    case "web":
    74→      return "fullstack-app";
    75→    case "android":
    76→      return "frontend-setup";
    77→    case "cli":
    78→      return "scaffold";
    79→    case "api":
    80→      return "backend-api";
    81→    default:
    82→      return undefined;
    83→  }
    84→}
    85→
    86→/**
    87→ * Client-side fetch of a SKILL.md content string from /api/skill-content.
    88→ * The API route reads the real file from /skills/<name>/SKILL.md via the
    89→ * server-only loader. Returns undefined on error or 404 (e.g. skill missing).
    90→ *
    91→ * This is how the orchestrator (which runs client-side) gets the SKILL.md
    92→ * content into the pipeline: the loader is server-only (uses fs), so we
    93→ * expose its content via an API route and fetch it here.
    94→ */
    95→export async function fetchSkillContent(skillName: string): Promise<string | undefined> {
    96→  try {
    97→    const res = await fetch(`/api/skill-content?name=${encodeURIComponent(skillName)}`);
    98→    if (!res.ok) return undefined;
    99→    const data = await res.json();
   100→    if (!data || typeof data.content !== "string") return undefined;
   101→    return data.content as string;
   102→  } catch {
   103→    return undefined;
   104→  }
   105→}
   106→
   107→/**
   108→ * Pre-fetch SKILL.md content for every candidate target kind in parallel.
   109→ * Returns a partial map { kind -> content }. Kinds with no matching skill
   110→ * (or fetch failures) are omitted from the map. The orchestrator passes
   111→ * this map to detectTargets() so the Decision Engine can boost policies
   112→ * whose `choose` value is mentioned in the skill text.
   113→ */
   114→export async function fetchSkillContentForCandidateKinds(
   115→  kinds: PlatformKind[]
   116→): Promise<Partial<Record<PlatformKind, string>>> {
   117→  const out: Partial<Record<PlatformKind, string>> = {};
   118→  await Promise.all(
   119→    kinds.map(async (kind) => {
   120→      const skillName = skillForTarget(kind);
   121→      if (!skillName) return;
   122→      const content = await fetchSkillContent(skillName);
   123→      if (content) out[kind] = content;
   124→    })
   125→  );
   126→  return out;
   127→}

   129→/** Detect generation targets (multi-target) from a prompt. */
   130→export function detectTargets(
   131→  prompt: string,
   132→  /**
   133→   * Optional map of { platformKind -> SKILL.md content }. When supplied,
   134→   * each target's pickStack() call passes the matching skill content to
   135→   * decisionEngine.decide() so the skill can boost policies it endorses.
   136→   * The store calls detectTargets(prompt) without this map (UI display
   137→   * only — no boost needed); the orchestrator calls it with the map after
   138→   * fetching skill content from /api/skill-content.
   139→   */
   140→  skillContentMap?: Partial<Record<PlatformKind, string>>
   141→): DetectedTargets[] {
   142→  const caps = detectCapabilities(prompt);
   143→  const nfs = detectNonFunctionals(prompt);
   144→  const p = " " + prompt.toLowerCase() + " ";
   145→  const out: DetectedTargets[] = [];
   146→  const wants = (re: RegExp) => re.test(p);
   147→
   148→  // Domain inference: if the prompt mentions CAD/3D/GPU/game but no explicit
   149→  // platform keyword, infer windows/native (not the default web). This prevents
   150→  // "Build CAD software" from falling through to a generic Next.js app.
   151→  const isCAD = /\b(cad|autocad|3d modeling|opengl|directx|vulkan|rendering)\b/.test(p);
   152→  const isGame = /\b(game|unity|godot|2d platformer|3d game)\b/.test(p);
   153→  const domainInferredWindows = isCAD && !wants(/\b(web|android|cli|api)\b/);
   154→  const domainInferredGame = isGame && !wants(/\b(web|windows|android|cli|api)\b/);
   155→
   156→  const multi =
   157→    [
   158→      wants(/\b(windows|desktop|winui|wpf|winforms|win32)\b/) || domainInferredWindows,
   159→      wants(/\b(android|mobile( app)?|kotlin|flutter|companion app|phone app)\b/),
   160→      wants(/\b(web( site| app| admin| portal)?|website|landing|marketing site|saas|portal)\b/),
   161→      wants(/\b(api|rest|backend service|microservice)\b/),
   162→      wants(/\b(cli|command.line|terminal tool)\b/),
   163→    ].filter(Boolean).length > 1;
   164→
   165→  if (wants(/\b(windows|desktop|winui|wpf|winforms|win32|\.net\s*(desktop|app))\b/) || domainInferredWindows) {
   166→    const kind = "windows" as PlatformKind;
   167→    const { stack, decision } = decisionEngine.pickStack(kind, prompt, caps, nfs, skillContentMap?.[kind]);
   168→    out.push({ kind, label: multi ? "Desktop App" : "App", role: multi ? "Primary desktop workspace" : "Windows desktop application", stack, capabilities: caps, policies: [decision] });
   169→  }
   170→  if (wants(/\b(android|mobile( app)?|kotlin|flutter|companion app|phone app)\b/)) {
   171→    const kind = "android" as PlatformKind;
   172→    const { stack, decision } = decisionEngine.pickStack(kind, prompt, caps, nfs, skillContentMap?.[kind]);
   173→    out.push({ kind, label: multi ? "Android Companion" : "App", role: multi ? "Mobile companion app" : "Android application", stack, capabilities: caps, policies: [decision] });
   174→  }
   175→  if (wants(/\b(web( site| app| admin| portal)?|website|landing|marketing site|saas|portal|browser)\b/)) {
   176→    const kind = "web" as PlatformKind;
   177→    const { stack, decision } = decisionEngine.pickStack(kind, prompt, caps, nfs, skillContentMap?.[kind]);
   178→    out.push({ kind, label: multi ? "Web Portal" : "App", role: multi ? "Web admin portal" : "Web application", stack, capabilities: caps, policies: [decision] });
   179→  }
   180→  if (wants(/\b(api|rest( api)?|graphql|backend service|microservice)\b/)) {
   181→    const kind = "api" as PlatformKind;
   182→    const { stack, decision } = decisionEngine.pickStack(kind, prompt, caps, nfs, skillContentMap?.[kind]);
   183→    out.push({ kind, label: "API Service", role: "Backend API and data layer", stack, capabilities: caps, policies: [decision] });
   184→  }
   185→  if (wants(/\b(cli|command.line|terminal tool|brew install|cargo install)\b/)) {
   186→    const kind = "cli" as PlatformKind;
   187→    const { stack, decision } = decisionEngine.pickStack(kind, prompt, caps, nfs, skillContentMap?.[kind]);
   188→    out.push({ kind, label: "CLI Tool", role: "Command-line utility", stack, capabilities: caps, policies: [decision] });
   189→  }
   190→  if (wants(/\b(ai agent|autonomous agent|assistant service|chatbot|support agent)\b/)) {
   191→    out.push({ kind: "api" as PlatformKind, label: "AI Agent", role: "Autonomous agent service", stack: "Python + LangGraph", capabilities: caps, policies: [decisionEngine.pickStack("api", prompt, caps, nfs, skillContentMap?.["api"]).decision] });
   192→  }
   193→  if (wants(/\b(library|sdk|npm package|crate)\b/)) {
   194→    out.push({ kind: "library" as PlatformKind, label: "Library", role: "Reusable library / SDK", stack: /\b(rust|crate)\b/.test(p) ? "Rust crate" : "TypeScript library", capabilities: caps, policies: [] });
   195→  }
   196→  if (domainInferredGame) {
   197→    out.push({ kind: "library" as PlatformKind, label: "Game", role: "Interactive game", stack: "Godot + GDScript", capabilities: caps, policies: [] });
   198→  }
   199→  if (out.length === 0) {
   200→    const kind = "web" as PlatformKind;
   201→    const { stack, decision } = decisionEngine.pickStack(kind, prompt, caps, nfs, skillContentMap?.[kind]);
   202→    out.push({ kind, label: "Web App", role: "Web application", stack, capabilities: caps, policies: [decision] });
   203→  }
   204→  return out;
   205→}

   207→export class Orchestrator {
   208→  /** Bootstrap: register all data into registries and wire the event bus. */
   209→  bootstrap(): void {
   210→    // registries are populated by the index module import side-effects;
   211→    // here we just wire observability to the execution engine event bus.
   212→    executionEngine.subscribe((e) => {
   213→      observability.recordEvent(e);
   214→    });
   215→  }
   216→
   217→  /**
   218→   * Run a build: select workflow, detect targets + capabilities, write
   219→   * requirements/decision memory, compile the workflow DAG, submit to the
   220→   * execution engine, and return the plan.
   221→   */
   222→  async startBuild(prompt: string, projectId?: string): Promise<OrchestrationResult & { tasks: Task[] }> {
   223→    executionEngine.reset();
   224→    checkpointManager.clear();
   225→    artifactRegistry.clear();
   226→
   227→    // Context Builder pulls relevant memory for the planner
   228→    const ctx = contextBuilder.buildForAgent("planner", { kinds: ["requirements", "decision", "conversation"], prompt });
   229→
   230→    // Capability Detection (runs first)
   231→    const capabilities = detectCapabilities(prompt);
   232→    observability.recordEvent({
   233→      id: `ev-${Date.now()}`,
   234→      ts: Date.now(),
   235→      type: "capability-detected",
   236→      message: `Detected capabilities: ${capabilities.join(", ") || "none"}`,
   237→      level: "info",
   238→    });
   239→
   240→    // Autonomy gate: Ambiguity Detection. If the requirement is too ambiguous
   241→    // (score > AMBIGUITY_THRESHOLD), emit a human-question and pause the
   242→    // workflow instead of inventing business requirements. The engine asks
   243→    // ONLY when information is missing, conflicting, or an external resource
   244→    // is needed without credentials.
   245→    const ambiguity = detectAmbiguity(prompt);
   246→    observability.recordEvent({
   247→      id: `ev-${Date.now()}-amb`,
   248→      ts: Date.now(),
   249→      type: "capability-detected",
   250→      message: `Ambiguity score ${ambiguity.score.toFixed(2)} (threshold ${AMBIGUITY_THRESHOLD})${ambiguity.shouldAsk ? " → asking user" : " → proceeding autonomously"}`,
   251→      level: ambiguity.shouldAsk ? "warn" : "info",
   252→    });
   253→    const pendingQuestion = askQuestionIfNeeded(prompt);
   254→
   255→    // ---- Phase A1: Pre-fetch SKILL.md content for every candidate target ----
   256→    // The orchestrator runs client-side, so it fetches the skill content via
   257→    // /api/skill-content (the server-only loader reads /skills/<name>/SKILL.md).
   258→    // The map is passed to detectTargets() so the Decision Engine can boost
   259→    // policies whose `choose` value is mentioned in the skill text, and to
   260→    // generateForTarget() so the generators can include the skill guidance as
   261→    // a comment header in their README.
   262→    const candidateKinds: PlatformKind[] = ["windows", "android", "web", "api", "cli"];
   263→    const skillContentMap = await fetchSkillContentForCandidateKinds(candidateKinds);
   264→    const skillFetchCount = Object.keys(skillContentMap).length;
   265→    if (skillFetchCount > 0) {
   266→      observability.recordEvent({
   267→        id: `ev-${Date.now()}-skill`,
   268→        ts: Date.now(),
   269→        type: "memory-written",
   270→        message: `Loaded ${skillFetchCount} SKILL.md file(s) from /api/skill-content: ${candidateKinds
   271→          .filter((k) => skillContentMap[k])
   272→          .map((k) => `${k}→${skillForTarget(k)}`)
   273→          .join(", ")}`,
   274→        level: "info",
   275→      });
   276→    }
   277→
   278→    // Detect multi-targets + decisions (with skill-content boost)
   279→    const targets = detectTargets(prompt, skillContentMap);
   280→    const decisions = targets.flatMap((t) => t.policies);
   281→
   282→    // Write Requirements Memory + Decision Memory
   283→    projectMemory.write("requirements", "Original Prompt", prompt, "user");
   284→    projectMemory.write("requirements", "Detected Targets", targets.map((t) => `${t.label}: ${t.stack} (${t.role})`).join("\n"), "decision-engine");
   285→    projectMemory.write("decision", "Stack Selection", decisions.map((d) => `${d.topic}: ${d.chosen} (${d.confidence})`).join("\n"), "decision-engine");
   286→    projectMemory.write("architecture", "Capabilities", capabilities.join(", ") || "none", "decision-engine");
   287→
   288→    // ---- Phase A2: Read Architecture + Decision Memory back into a context ----
   289→    // The orchestrator just wrote to Architecture + Decision Memory above. We
   290→    // read those records back and concatenate them into a single memoryContext
   291→    // string that is passed to every generator so the generated project
   292→    // carries the engine's decision history in its README.
   293→    const archMemory = projectMemory.read("architecture");
   294→    const decisionMemory = projectMemory.read("decision");
   295→    const memoryContext =
   296→      [
   297→        "=== ARCHITECTURE MEMORY ===",
   298→        ...archMemory.map((r) => `[${r.title}] (v${r.version}, source: ${r.source})\n${r.content}`),
   299→        "",
   300→        "=== DECISION MEMORY ===",
   301→        ...decisionMemory.map((r) => `[${r.title}] (v${r.version}, source: ${r.source})\n${r.content}`),
   302→      ].join("\n");
   303→
   304→    // Workflow selection
   305→    const workflow = workflowEngine.select(prompt);
   306→    observability.recordEvent({
   307→      id: `ev-${Date.now()}-wf`,
   308→      ts: Date.now(),
   309→      type: "workflow-selected",
   310→      workflowId: workflow.id,
   311→      message: `Selected workflow: ${workflow.name}`,
   312→      level: "info",
   313→    });
   314→
   315→    // ---- Real generation: invoke the generator for each detected target ----
   316→    // The Desktop Generator (Anvil) produces WinUI 3 / Tauri scaffolding; the
   317→    // Android Generator (Droid) produces Kotlin+Compose / Flutter; the Web
   318→    // Generator (Forge) produces a REAL compilable Next.js app (Prisma + auth
   319→    // + CRUD pages derived from the requirement's data model). Files are
   320→    // versioned into the Artifact Registry and emitted as artifact-produced
   321→    // events for observability.
   322→    //
   323→    // Phase B: each target's generation is routed through the AgentRuntime,
   324→    // not by calling generateForTarget() directly. The orchestrator:
   325→    //   1. Looks up the PlatformAdapter for the target kind via the registry.
   326→    //   2. Uses adapter.generators[0] as the primary generator agent role.
   327→    //   3. Looks up the Agent for that role via the agent registry.
   328→    //   4. Constructs an AgentRuntime(agent) and calls runtime.execute(task, ctx).
   329→    // The runtime wraps the generator call with:
   330→    //   - a task-started event attributing the work to the agent
   331→    //   - reading Architecture + Decision memory (built into memoryContext)
   332→    //   - the role → generator-function map (desktop→WinUI3App, etc.)
   333→    //   - a task-succeeded/failed event with the result
   334→    // The adapter's defaultStack + capabilities + packagingTools are passed
   335→    // through to the generator as context (the generators receive them but
   336→    // their output is unchanged — they're for attribution/downstream tooling).
   337→    //
   338→    // Phase A1 + A2: pass the per-target skillContent (from SKILL.md) and
   339→    // the memoryContext (Architecture + Decision memory) into each generator
   340→    // so they flow through the pipeline into the generated README.
   341→    const generationResults: GenerationResult[] = [];
   342→    for (let i = 0; i < targets.length; i++) {
   343→      const t = targets[i];
   344→      const targetId = `t${i + 1}`;
   345→      const projectName = promptToName(prompt) || `App${i + 1}`;
   346→
   347→      // ---- Phase B2: Route through the Platform Adapter Registry ----
   348→      // Look up the adapter for this target kind. The adapter declares the
   349→      // generators, packagingTools, capabilities, defaultStack, and
   350→      // requiredPermissions for the platform.
   351→      const adapter = registries.platformAdapters.get(t.kind);
   352→      if (!adapter) {
   353→        throw new Error(`No platform adapter registered for kind: ${t.kind}`);
   354→      }
   355→      // The first generator in the adapter's generators list is the primary
   356→      // generator agent for this platform (e.g. "desktop-generator" for
   357→      // windows, "frontend-generator" for web, "android-generator" for
   358→      // android, "backend-generator" for cli/api/library).
   359→      const generatorAgentRole = adapter.generators[0] as AgentRole;
   360→      const agent = registries.agents.get(generatorAgentRole);
   361→      if (!agent) {
   362→        throw new Error(`No agent registered for role: ${generatorAgentRole}`);
   363→      }
   364→
   365→      // Construct a Task representing this generation work — for attribution
   366→      // in the observability timeline and the execution engine event bus.
   367→      // The task is NOT submitted to the execution engine (the orchestrator
   368→      // runs generation synchronously inline); it's purely a context object
   369→      // for the AgentRuntime's attribution events.
   370→      const genTask = makeTask({
   371→        workflowId: workflow.id,
   372→        stageId: "generate",
   373→        title: `Generate ${t.label}`,
   374→        description: `${agent.name} (${agent.role}) generates ${t.stack} code for ${t.label}`,
   375→        agent: agent.role,
   376→        dependsOn: [],
   377→      });
   378→
   379→      // Run the generation through the AgentRuntime so the work is
   380→      // attributed to the agent and the runtime reads Architecture + Decision
   381→      // Memory before calling the generator (matching the spec: the runtime
   382→      // wraps generateForTarget, it doesn't replace it).
   383→      const runtime = new AgentRuntime(agent);
   384→      const runtimeResult = await runtime.execute(genTask, {
   385→        skillContent: skillContentMap[t.kind],
   386→        memoryContext,
   387→        projectPath: projectId ?? `proj-${Date.now()}`,
   388→        targetId,
   389→        prompt,
   390→        capabilities,
   391→        nonFunctionals: detectNonFunctionals(prompt),
   392→        projectName,
   393→        stack: t.stack,
   394→        adapter,
   395→      });
   396→
   397→      if (!runtimeResult.success || !runtimeResult.result) {
   398→        // Surface the runtime's error to the caller — the build fails fast
   399→        // rather than continuing with partial output.
   400→        throw new Error(runtimeResult.error ?? `Generation failed for ${t.label}`);
   401→      }
   402→
   403→      const result = runtimeResult.result;
   404→      observability.recordEvent({
   405→        id: `ev-${Date.now()}-gen-${i}`,
   406→        ts: Date.now(),
   407→        type: "artifact-produced",
   408→        stageId: "generate",
   409→        message: `${t.label}: ${agent.name} (${agent.role}) generated ${result.files.length} files (${result.stack}) via ${adapter.name} adapter${skillContentMap[t.kind] ? " + skill content embedded" : ""}${memoryContext ? " + memory context embedded" : ""}`,
   410→        level: "success",
   411→      });
   412→      projectMemory.write(
   413→        "code",
   414→        `${t.label} source`,
   415→        result.files.map((f) => `${f.path} (${(f.content.length / 1024).toFixed(1)} KB)`).join("\n"),
   416→        result.producedBy
   417→      );
   418→      generationResults.push(result);
   419→    }
   420→    let totalFiles = generationResults.reduce((n, g) => n + g.files.length, 0);
   421→    const wsProjectId = projectId ?? `proj-${Date.now()}`;
   422→
   423→    // ---- Tier 1 Skills: generate tests, CI/CD, Docker ----
   424→    // These are REAL skills — each produces compilable/validatable files.
   425→    const tier1Ctx = { projectName: promptToName(prompt) || "App", prompt, capabilities, nonFunctionals: detectNonFunctionals(prompt) };
   426→    const tier1Files: { folder: string; files: import("./generators").VirtualFile[] }[] = [];
   427→
   428→    // 1. Test generation — web (Vitest), desktop (xUnit), android (JUnit)
   429→    for (const t of targets) {
   430→      if (t.kind === "web") tier1Files.push({ folder: "web-admin", files: generateWebTests(tier1Ctx) });
   431→      if (t.kind === "windows") tier1Files.push({ folder: "desktop", files: generateDesktopTests(tier1Ctx) });
   432→      if (t.kind === "android") tier1Files.push({ folder: "android", files: generateAndroidTests(tier1Ctx) });
   433→    }
   434→
   435→    // 2. CI/CD generation — .github/workflows/build.yml
   436→    const cicdFiles = generateCICD(promptToName(prompt) || "App", targets.map((t) => ({ kind: t.kind, label: t.label, role: t.role, stack: t.stack, id: "" })));
   437→    tier1Files.push({ folder: "", files: cicdFiles });
   438→
   439→    // 3. Docker generation — Dockerfile per target + docker-compose.yml
   440→    const dockerFiles = generateDocker(promptToName(prompt) || "App", targets.map((t) => ({ kind: t.kind, label: t.label, role: t.role, stack: t.stack, id: "" })));
   441→    tier1Files.push({ folder: "", files: dockerFiles });
   442→
   443→    // Materialize Tier 1 files to workspace
   444→    for (const { folder, files } of tier1Files) {
   445→      if (files.length === 0) continue;
   446→      try {
   447→        await fetch("/api/workspace", {
   448→          method: "POST",
   449→          headers: { "Content-Type": "application/json" },
   450→          body: JSON.stringify({ projectId: wsProjectId, targetFolder: folder || "root", files }),
   451→        });
   452→      } catch { /* best effort */ }
   453→      totalFiles += files.length;
   454→    }
   455→    observability.recordEvent({
   456→      id: `ev-${Date.now()}-tier1`,
   457→      ts: Date.now(),
   458→      type: "artifact-produced",
   459→      stageId: "generate",
   460→      message: `Tier 1: generated ${tier1Files.reduce((n, f) => n + f.files.length, 0)} test/cicd/docker files`,
   461→      level: "success",
   462→    });
   463→
   464→    // Compile DAG
   465→    const tasks = workflowEngine.compile(workflow, workflow.stages.map((s) => s.id as StageId));
   466→
   467→    // Materialize generated files to a real on-disk workspace so the
   468→    // compilation gate can run `tsc --noEmit` against them.
   469→    const workspacePaths: Record<string, string> = {}; // targetId -> path
   470→    for (let i = 0; i < targets.length; i++) {
   471→      const t = targets[i];
   472→      const gen = generationResults[i];
   473→      const folder = t.kind === "windows" ? "desktop" : t.kind === "android" ? "android" : t.kind === "web" ? "web-admin" : t.kind === "cli" ? "cli" : "app";
   474→      try {
   475→        const res = await fetch("/api/workspace", {
   476→          method: "POST",
   477→          headers: { "Content-Type": "application/json" },
   478→          body: JSON.stringify({ projectId: wsProjectId, targetFolder: folder, files: gen.files }),
   479→        });
   480→        if (res.ok) {
   481→          const { path: wsPath } = await res.json();
   482→          workspacePaths[`t${i + 1}`] = wsPath;
   483→          observability.recordEvent({
   484→            id: `ev-${Date.now()}-ws-${i}`,
   485→            ts: Date.now(),
   486→            type: "artifact-produced",
   487→            stageId: "generate",
   488→            message: `${t.label}: materialized ${gen.files.length} files to ${wsPath}`,
   489→            level: "info",
   490→          });
   491→        }
   492→      } catch {
   493→        // workspace write is best-effort; gates will skip if no workspace
   494→      }
   495→    }
   496→    // Annotate the compilation gate task with the web workspace (runs tsc).
   497→    // For multi-target builds, attach the primary target's workspace + type.
   498→    // Desktop/android static validation happens via their own gate tasks below.
   499→    const webWorkspace = workspacePaths["t1"] ?? Object.values(workspacePaths)[0] ?? undefined;
   500→    const primaryTarget = targets[0];
   501→    const primaryType = primaryTarget?.kind === "windows" ? "desktop" : primaryTarget?.kind === "android" ? "android" : "web";
   502→    for (const task of tasks) {
   503→      // Set gateContext for ALL gate tasks. Structural gates (architecture,
   504→      // security, etc.) need artifactCount to pass. Compilation gates need
   505→      // workspacePath + targetType.
   506→      if (task.gate) {
   507→        (task as Task & { gateContext?: import("./self-healing").GateEvaluationContext }).gateContext = {
   508→          workspacePath: webWorkspace,
   509→          artifactCount: totalFiles,
   510→          targetType: task.gate === "compilation" ? primaryType : undefined,
   511→        };
   512→      }
   513→      // Attach toolId + cwd to the build stage task so it runs npm-build (web only)
   514→      if (task.stageId === "build" && !task.gate && webWorkspace && primaryTarget?.kind === "web") {
   515→        task.toolId = "npm-build";
   516→        (task as Task & { args?: { cwd?: string } }).args = { cwd: webWorkspace };
   517→      }
   518→    }
   519→
   520→    // For multi-target builds, add per-target compilation gate tasks for
   521→    // desktop/android so their static validators run too.
   522→    for (let i = 0; i < targets.length; i++) {
   523→      const t = targets[i];
   524→      const wsPath = workspacePaths[`t${i + 1}`];
   525→      if (!wsPath) continue;
   526→      const tType = t.kind === "windows" ? "desktop" : t.kind === "android" ? "android" : null;
   527→      if (!tType || tType === primaryType) continue; // skip, already covered
   528→      const extraGate = makeTask({
   529→        workflowId: workflow.id,
   530→        stageId: "build",
   531→        title: `Gate: compilation (${tType})`,
   532→        description: `Static validation for ${t.label}`,
   533→        agent: "orchestrator" as AgentRole,
   534→        dependsOn: [],
   535→        gate: "compilation" as GateId,
   536→      });
   537→      (extraGate as Task & { gateContext?: import("./self-healing").GateEvaluationContext }).gateContext = {
   538→        workspacePath: wsPath,
   539→        artifactCount: totalFiles,
   540→        targetType: tType as "desktop" | "android",
   541→      };
   542→      tasks.push(extraGate);
   543→    }
   544→
   545→    // Token budget check
   546→    if (!tokenBudgetManager.withinBudget("planner", workflow.id)) {
   547→      observability.recordEvent({
   548→        id: `ev-${Date.now()}-budget`,
   549→        ts: Date.now(),
   550→        type: "task-failed",
   551→        message: "Token budget exhausted for planner",
   552→        level: "error",
   553→      });
   554→    }
   555→    // Token charging: real tokens come from the z-ai SDK chat response
   556→    // (captured in use-chat.ts via usage.total_tokens). For non-LLM tasks
   557→    // (planner context, generator file production), tokens = 0 — no estimate.
   558→    // The ctx.tokenEstimate is NOT charged — it was only used for budget
   559→    // checking, not for observability metrics.
   560→    tokenBudgetManager.charge("planner", workflow.id, 0);
   561→    // Generator agents produce files via deterministic templates, not LLM
   562→    // calls — no tokens consumed. Real tokens are only from /api/chat.
   563→    for (const g of generationResults) {
   564→      observability.chargeTokens(g.producedBy, 0, workflow.id);
   565→      tokenBudgetManager.charge(g.producedBy, workflow.id, 0);
   566→    }
   567→
   568→    // Submit to execution engine (parallel, dependency-scheduled). If the
   569→    // ambiguity gate raised a question, askQuestionIfNeeded already cancelled
   570→    // running tasks; we still return the plan so the UI can surface the question.
   571→    executionEngine.submitAll(tasks);
   572→    if (pendingQuestion) {
   573→      // Pause: cancel anything that just started so the engine waits for the
   574→      // user's clarification before resuming.
   575→      executionEngine.cancelAll();
   576→      projectMemory.write("requirements", "Pending Question", pendingQuestion, "ambiguity-detector");
   577→    }
   578→
   579→    return {
   580→      workflow,
   581→      targets,
   582→      decisions,
   583→      capabilities,
   584→      tasks,
   585→      generatedFiles: totalFiles,
   586→      ambiguityScore: ambiguity.score,
   587→      pendingQuestion,
   588→    };
   589→  }

   591→  /** Checkpoint after a stage completes (for recovery). */
   592→  checkpoint(stageId: StageId, workflowId: WorkflowId, snapshot: Record<string, import("./types").TaskStatus>): void {
   593→    const cp = checkpointManager.save(stageId, workflowId, snapshot, projectMemory.version());
   594→    observability.recordEvent({
   595→      id: `ev-${Date.now()}-cp`,
   596→      ts: Date.now(),
   597→      type: "checkpoint-saved",
   598→      stageId,
   599→      message: `Checkpoint saved at ${stageId}`,
   600→      level: "debug",
   601→    });
   602→    void cp;
   603→  }
   604→
   605→  /** Resume from the last checkpoint after a crash. */
   606→  resume(): { stageId: string; snapshot: Record<string, import("./types").TaskStatus> } | null {
   607→    const r = checkpointManager.resume();
   608→    if (r) {
   609→      observability.recordEvent({
   610→        id: `ev-${Date.now()}-cr`,
   611→        ts: Date.now(),
   612→        type: "checkpoint-restored",
   613→        message: `Resumed from checkpoint at ${r.stageId}`,
   614→        level: "info",
   615→      });
   616→    }
   617→    return r;
   618→  }

   632→/** Derive a clean project name from the prompt (shared with the store). */
   633→function promptToName(prompt: string): string {
   634→  let trimmed = prompt.trim().toLowerCase();
   635→  let prev = "";
   636→  while (prev !== trimmed) {
   637→    prev = trimmed;
   638→    trimmed = trimmed.replace(
   639→      /^(build|build me|create|make|generate|develop|i want|i need|please|a|an|the|me|some)\s+/i,
   640→      ""
   641→    );
   642→  }
   643→  const stopwords = new Set([
   644→    "in", "that", "with", "for", "to", "and", "of", "on", "using", "via",
   645→    "which", "from", "into", "where", "when", "app", "application", "as",
   646→    "companion", "portal", "admin", "me",
   647→  ]);
   648→  const acronyms = new Set(["cli", "ai", "api", "sdk", "saas", "ui", "ux", "ios", "ml", "crm", "cms", "erp", "hrm"]);
   649→  const words: string[] = [];
   650→  for (const w of trimmed.split(/\s+/).filter(Boolean)) {
   651→    if (stopwords.has(w)) break;
   652→    words.push(w);
   653→    if (words.length >= 3) break;
   654→  }
   655→  if (words.length === 0) return "App";
   656→  return words
   657→    .map((w) => (acronyms.has(w) ? w.toUpperCase() : w.charAt(0).toUpperCase() + w.slice(1)))
   658→    .join(" ")
   659→    .replace(/[^a-zA-Z0-9 ]/g, "")
   660→    .trim();
   661→}
